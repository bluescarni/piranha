# Add the unit test library.
ADD_DEFINITIONS(-DBOOST_TEST_DYN_LINK)

MACRO(ADD_PIRANHA_TESTCASE arg1)
	IF(CMAKE_BUILD_TYPE STREQUAL "Debug")
		ADD_EXECUTABLE(${arg1} ${arg1}.cpp)
		TARGET_LINK_LIBRARIES(${arg1} piranha_static ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY})
		ADD_TEST(${arg1} ${arg1})
	ENDIF(CMAKE_BUILD_TYPE STREQUAL "Debug")
ENDMACRO(ADD_PIRANHA_TESTCASE)

MACRO(ADD_PIRANHA_PERFORMANCE_TESTCASE arg1)
	IF(CMAKE_BUILD_TYPE STREQUAL "Release")
		ADD_EXECUTABLE("${arg1}_perf" ${arg1}.cpp)
		TARGET_LINK_LIBRARIES("${arg1}_perf" piranha_static ${Boost_UNIT_TEST_FRAMEWORK_LIBRARY})
		ADD_TEST("${arg1}_perf" "${arg1}_perf")
	ENDIF(CMAKE_BUILD_TYPE STREQUAL "Release")
ENDMACRO(ADD_PIRANHA_PERFORMANCE_TESTCASE)
