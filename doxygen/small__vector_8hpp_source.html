58 // This is essentially a reduced std::vector replacement that should use less storage on most implementations
59 // (e.g., it has a size of 16 on Linux 64-bit and it _should_ have a size of 8 on many 32-bit archs).
60 // NOTE: earlier versions of this class used to have support for custom allocation. See, e.g., commit
61 // 6e1dd235c729ed05eb6e872dceece2f20a624f32
62 // It seems however that there are some defects currently in the standard C++ library that make it quite hard
63 // to implement reasonably strong exception safety and nothrow guarantees. See, e.g.,
64 // http://cplusplus.github.io/LWG/lwg-defects.html#2103
65 // It seems like in the current standard it is not possible to implemement noexcept move assignment not even
66 // for std::alloc. The crux of the matter is that if propagate_on_container_move_assignment is not defined,
67 // we will need in general a memory allocation (that can fail).
68 // Another, less important, problem is that for copy assignment the presence of
69 // propagate_on_container_copy_assignment seems to force one to nuke pre-emptively the content of the container
70 // before doing anything (thus preventing strong forms of exception safety).
71 // The approach taken here is thus to just hard-code std::allocator, which is stateless, and force nothrow
72 // behaviour when doing assignments. This should work with all stateless allocators (including our own allocator).
73 // Unfortunately it is not clear if the extension allocators from GCC are stateless, so better not to muck
74 // with them (apart from experimentation).
75 // See also:
76 // http://stackoverflow.com/questions/12332772/why-arent-container-move-assignment-operators-noexcept
77 // NOTE: now we do not use the allocator at all, as we cannot guarantee it has a standard layout. Keep the comments
78 // above as an historical reference, might come useful in the future :)
79 template <typename T>
80 class dynamic_storage
81 {
82  PIRANHA_TT_CHECK(is_container_element,T);
83  public:
84  using size_type = unsigned char;
85  using value_type = T;
86  private:
87  using pointer = value_type *;
88  using const_pointer = value_type const *;
89  // NOTE: this bit of TMP is to avoid checking an always-false condition on reserve() on most platforms, which triggers a compiler
90  // warning on GCC 4.7.
91  static const std::size_t max_alloc_size = std::numeric_limits<std::size_t>::max() / sizeof(value_type);
92  static const bool need_reserve_check = std::numeric_limits<size_type>::max() > max_alloc_size;
93  static bool reserve_check_size(const size_type &, const std::false_type &)
94  {
95  return false;
96  }
97  static bool reserve_check_size(const size_type &new_capacity, const std::true_type &)
98  {
99  return new_capacity > max_alloc_size;
100  }
101  public:
102  static const size_type max_size = std::numeric_limits<size_type>::max();
103  using iterator = pointer;
104  using const_iterator = const_pointer;
105  dynamic_storage() : m_tag(0u),m_size(0u),m_capacity(0u),m_ptr(nullptr) {}
106  dynamic_storage(dynamic_storage &&other) noexcept : m_tag(0u),m_size(other.m_size),m_capacity(other.m_capacity),m_ptr(other.m_ptr)
107  {
108  // Erase the other.
109  other.m_size = 0u;
110  other.m_capacity = 0u;
111  other.m_ptr = nullptr;
112  }
113  dynamic_storage(const dynamic_storage &other) :
114  m_tag(0u),m_size(0u),m_capacity(other.m_size), // NOTE: when copying, we set the capacity to the same value of the size.
115  m_ptr(nullptr)
116  {
117  // Obtain storage. Will just return nullptr if requested size is zero.
118  m_ptr = allocate(other.m_size);
119  // Attempt to copy-construct the elements from other.
120  try {
121  for (; m_size < other.m_size; ++m_size) {
122  construct(m_ptr + m_size,other[m_size]);
123  }
124  } catch (...) {
125  // Roll back the construction and deallocate before re-throwing.
126  destroy_and_deallocate();
127  throw;
128  }
129  }
130  ~dynamic_storage()
131  {
132  // NOTE: here we should replace with bidirectional tt, if we ever implement it.
133  PIRANHA_TT_CHECK(is_forward_iterator,iterator);
134  PIRANHA_TT_CHECK(is_forward_iterator,const_iterator);
135  piranha_assert(m_tag == 0u);
136  destroy_and_deallocate();
137  }
138  dynamic_storage &operator=(dynamic_storage &&other) noexcept
139  {
140  if (likely(this != &other)) {
141  // Destroy and deallocate this.
142  destroy_and_deallocate();
143  // Just pilfer the resources.
144  m_size = other.m_size;
145  m_capacity = other.m_capacity;
146  m_ptr = other.m_ptr;
147  // Nuke the other.
148  other.m_size = 0u;
149  other.m_capacity = 0u;
150  other.m_ptr = nullptr;
151  }
152  return *this;
153  }
154  dynamic_storage &operator=(const dynamic_storage &other)
155  {
156  if (likely(this != &other)) {
157  dynamic_storage tmp(other);
158  *this = std::move(tmp);
159  }
160  return *this;
161  }

206  void reserve(const size_type &new_capacity)
207  {
208  piranha_assert(consistency_checks());
209  // No need to do anything if we already have enough capacity.
210  if (new_capacity <= m_capacity) {
211  return;
212  }
213  // Check that we are not asking too much.
214  // NOTE: the first check is redundant right now, just keep it around in case max_size changes in the future.
215  if (unlikely(/*new_capacity > max_size ||*/ reserve_check_size(new_capacity,std::integral_constant<bool,need_reserve_check>()))) {
216  piranha_throw(std::bad_alloc,);
217  }
218  // Start by allocating the new storage. New capacity is at least one at this point.
219  pointer new_storage = allocate(new_capacity);
220  assert(new_storage != nullptr);
221  // Move in existing elements. Consistency checks ensure
222  // that m_size is not greater than m_capacity and, by extension, new_capacity.
223  for (size_type i = 0u; i < m_size; ++i) {
224  construct(new_storage + i,std::move((*this)[i]));
225  }
226  // Destroy and deallocate original content.
227  destroy_and_deallocate();
228  // Move in the new pointer and capacity.
229  m_capacity = new_capacity;
230  m_ptr = new_storage;
231  }
232  size_type capacity() const
233  {
234  return m_capacity;
235  }
236  std::size_t hash() const
237  {
238  return detail::vector_hasher(*this);
239  }
240  void resize(const size_type &new_size)
241  {
242  // NOTE: another redundant size check at the moment.
243  // if (unlikely(new_size > max_size)) {
244  // piranha_throw(std::bad_alloc,);
245  // }
246  if (new_size == m_size) {
247  return;
248  }
249  // The storage we are going to operate on is either the old one, if it has enough capacity,
250  // or new storage.
251  const bool new_storage = (m_capacity < new_size);
252  pointer storage = new_storage ? allocate(new_size) : m_ptr;
253  // NOTE: storage cannot be nullptr:
254  // - if new storage, new_size has to be at least 1 (new_size > m_capacity);
255  // - if not new storage, new_size <= m_capacity; m_ptr can be null only if capacity is 0, but then
256  // m_size is zero and new_size is 0 as well, and the function never arrived here because of the check above.
257  piranha_assert(storage != nullptr);
258  // Default-construct excess elements. We need to do this regardless of where the storage is coming from.
259  // This is also the only place we care about exception handling.
260  size_type i = m_size;
261  try {
262  for (; i < new_size; ++i) {
263  construct(storage + i);
264  }
265  } catch (...) {
266  // Roll back and dealloc.
267  for (size_type j = m_size; j < i; ++j) {
268  destroy(storage + j);
269  }
270  deallocate(storage);
271  throw;
272  }
273  // NOTE: no more exceptions thrown after this point.
274  if (new_storage) {
275  // Move in old elements into the new storage. As we had to increase the capacity,
276  // we know that new_size has to be greater than the old one, hence all old elements
277  // need to be moved over.
278  for (size_type j = 0u; j < m_size; ++j) {
279  construct(storage + j,std::move((*this)[j]));
280  }
281  // Erase the old content and assign new.
282  destroy_and_deallocate();
283  m_capacity = new_size;
284  m_ptr = storage;
285  } else {
286  // Destroy excess elements in the old storage.
287  for (size_type j = new_size; j < m_size; ++j) {
288  destroy(storage + j);
289  }
290  }
291  // In any case, we need to update the size.
292  m_size = new_size;
293  }
294  // Erase element. See the cooresponding code in static_vector.
295  iterator erase(const_iterator it)
296  {
297  piranha_assert(it != end());
298  auto retval = const_cast<iterator>(it);
299  const auto it_f = end() - 1;
300  for (; it != it_f; ++it) {
301  it->~T();
302  ::new (static_cast<void *>(const_cast<iterator>(it))) value_type(std::move(*(it + 1)));
303  }
304  it->~T();
305  m_size = static_cast<size_type>(m_size - 1u);
306  return retval;
307  }
308  private:
309  template <typename ... Args>
310  static void construct(pointer p, Args && ... args)
311  {
312  ::new (static_cast<void *>(p)) value_type(std::forward<Args>(args)...);
313  }
314  static void destroy(pointer p)
315  {
316  p->~value_type();
317  }
318  // Obtain new storage, and throw an error in case something goes wrong.
319  // NOTE: no need to check for zero, will aready return nullptr in that case.
320  static pointer allocate(const size_type &s)
321  {
322  return static_cast<pointer>(aligned_palloc(0u,static_cast<std::size_t>(s * sizeof(value_type))));
323  }
324  static void deallocate(pointer p)
325  {
326  aligned_pfree(0u,p);
327  }
328  // Common implementation of push_back().
329  template <typename U>
330  void push_back_impl(U &&x)
331  {
332  piranha_assert(consistency_checks());
333  if (unlikely(m_capacity == m_size)) {
334  increase_capacity();
335  }
336  construct(m_ptr + m_size,std::forward<U>(x));
337  ++m_size;
338  }
339  void destroy_and_deallocate()
340  {
341  piranha_assert(consistency_checks());
342  for (size_type i = 0u; i < m_size; ++i) {
343  // NOTE: could use POD optimisations here in principle.
344  destroy(m_ptr + i);
345  }
346  // NOTE: no need to check for nullptr, aligned_pfree already does it.
347  deallocate(m_ptr);
348  }
349  // Will try to double the capacity, or, in case this is not possible,
350  // will set the capacity to max_size. If the initial capacity is already max,
351  // then will throw an error.
352  void increase_capacity()
353  {
354  if (unlikely(m_capacity == max_size)) {
355  piranha_throw(std::bad_alloc,);
356  }
357  // NOTE: capacity should double, but without going past max_size, and in case it is zero it should go to 1.
358  const size_type new_capacity = (m_capacity > max_size / 2u) ? max_size : ((m_capacity != 0u) ?
359  static_cast<size_type>(m_capacity * 2u) : static_cast<size_type>(1u));
360  reserve(new_capacity);
361  }
362  bool consistency_checks()
363  {
364  // Size cannot be greater than capacity.
365  if (unlikely(m_size > m_capacity)) {
366  return false;
367  }
368  // If we allocated something, capacity cannot be zero.
369  if (unlikely(m_ptr != nullptr && m_capacity == 0u)) {
370  return false;
371  }
372  // If we have nothing allocated, capacity must be zero.
373  if (unlikely(m_ptr == nullptr && m_capacity != 0u)) {
374  return false;
375  }
376  return true;
377  }
378  private:
379  unsigned char m_tag;
380  size_type m_size;
381  size_type m_capacity;
382  pointer m_ptr;
383 };

419 // NOTE: here we are using a special rule from 9.2. If the union is standard-layout, and it contains two or more
420 // standard-layout classes, it will be possible to inspect the value of the common initial part of any of them. That is,
421 // we can access the initial m_tag member of the static and dynamic storage via both st and dy, and use it to detect
422 // which class of the union is active.
423 // See also:
424 // http://stackoverflow.com/questions/18564497/writing-into-the-last-byte-of-a-class
425 // http://www.informit.com/guides/content.aspx?g=cplusplus&seqNum=556
426 // http://en.wikipedia.org/wiki/C%2B%2B11#Unrestricted_unions
427 template <typename T, typename S>
428 union small_vector_union
429 {
430  using s_storage = typename std::conditional<S::value == 0u,static_vector<T,auto_static_size<T>::value>,static_vector<T,S::value>>::type;
431  using d_storage = dynamic_storage<T>;
432  // NOTE: each constructor must be invoked explicitly.
433  small_vector_union():m_st() {}
434  small_vector_union(const small_vector_union &other)
435  {
436  if (other.is_static()) {
437  ::new (static_cast<void *>(&m_st)) s_storage(other.g_st());
438  } else {
439  ::new (static_cast<void *>(&m_dy)) d_storage(other.g_dy());
440  }
441  }
442  small_vector_union(small_vector_union &&other) noexcept
443  {
444  if (other.is_static()) {
445  ::new (static_cast<void *>(&m_st)) s_storage(std::move(other.g_st()));
446  } else {
447  ::new (static_cast<void *>(&m_dy)) d_storage(std::move(other.g_dy()));
448  }
449  }
450  ~small_vector_union()
451  {
452  PIRANHA_TT_CHECK(std::is_standard_layout,s_storage);
453  PIRANHA_TT_CHECK(std::is_standard_layout,d_storage);
454  PIRANHA_TT_CHECK(std::is_standard_layout,small_vector_union);
455  if (is_static()) {
456  g_st().~s_storage();
457  } else {
458  g_dy().~d_storage();
459  }
460  }
461  small_vector_union &operator=(small_vector_union &&other) noexcept
462  {
463  if (unlikely(this == &other)) {
464  return *this;
465  }
466  if (is_static() == other.is_static()) {
467  if (is_static()) {
468  g_st() = std::move(other.g_st());
469  } else {
470  g_dy() = std::move(other.g_dy());
471  }
472  } else {
473  if (is_static()) {
474  // static vs dynamic.
475  // Destroy static.
476  g_st().~s_storage();
477  // Move construct dynamic from other.
478  ::new (static_cast<void *>(&m_dy)) d_storage(std::move(other.g_dy()));
479  } else {
480  // dynamic vs static.
481  g_dy().~d_storage();
482  ::new (static_cast<void *>(&m_st)) s_storage(std::move(other.g_st()));
483  }
484  }
485  return *this;
486  }
487  small_vector_union &operator=(const small_vector_union &other)
488  {
489  if (likely(this != &other)) {
490  *this = small_vector_union(other);
491  }
492  return *this;
493  }
494  bool is_static() const
495  {
496  return static_cast<bool>(m_st.m_tag);
497  }
498  // Getters.
499  const s_storage &g_st() const
500  {
501  piranha_assert(is_static());
502  return m_st;
503  }
504  s_storage &g_st()
505  {
506  piranha_assert(is_static());
507  return m_st;
508  }
509  const d_storage &g_dy() const
510  {
511  piranha_assert(!is_static());
512  return m_dy;
513  }
514  d_storage &g_dy()
515  {
516  piranha_assert(!is_static());
517  return m_dy;
518  }
519  s_storage m_st;
520  d_storage m_dy;
521 };

550 // NOTE: some possible improvements:
551 // - the m_size member of dynamic and static could be made a signed integer, the sign establishing the storage type
552 // and drop the m_tag member. This in principle would allow to squeeze some extra space from the static vector
553 // but not sure this is worth it;
554 // - POD optimisations in dynamic storage;
555 // - in the dynamic storage, it looks like on 64bit we can bump up the size member to 16 bit without changing size,
556 // thus we could sture ~16000 elements. BUT on 32bit this will change the size.
557 template <typename T, typename S = std::integral_constant<std::size_t,0u>>
558 class small_vector
559 {
560  PIRANHA_TT_CHECK(is_container_element,T);
561  PIRANHA_TT_CHECK(detail::check_integral_constant,S);
562  using u_type = detail::small_vector_union<T,S>;
563  using s_storage = typename u_type::s_storage;
564  using d_storage = typename u_type::d_storage;
565  // The size type will be the one with most range among the two storages.
566  using size_type_impl = max_int<typename s_storage::size_type,typename d_storage::size_type>;
567  template <typename U>
568  static constexpr U get_max(const U &a, const U &b)
569  {
570  return (a > b) ? a : b;
571  }
572  public:
574  static const typename std::decay<decltype(s_storage::max_size)>::type max_static_size = s_storage::max_size;
576  static const typename std::decay<decltype(d_storage::max_size)>::type max_dynamic_size = d_storage::max_size;
578  using size_type = size_type_impl;
579  private:
580  // If the size type can assume values larger than d_storage::max_size, then we need to run a check in resize().
581  static const bool need_resize_check = std::numeric_limits<size_type>::max() > d_storage::max_size;
582  static bool resize_check_size(const size_type &, const std::false_type &)
583  {
584  return false;
585  }
586  static bool resize_check_size(const size_type &size, const std::true_type &)
587  {
588  return size > d_storage::max_size;
589  }
590  public:
592  static const size_type max_size = get_max<size_type>(max_static_size,max_dynamic_size);
594  using value_type = T;
596  using iterator = value_type *;
598  using const_iterator = value_type const *;
599  private:
600  // NOTE: here we should replace with bidirectional tt, if we ever implement it.
601  PIRANHA_TT_CHECK(is_forward_iterator,iterator);
602  PIRANHA_TT_CHECK(is_forward_iterator,const_iterator);
603  // Enabler for ctor from init list.
604  template <typename U>
605  using init_list_enabler = typename std::enable_if<has_safe_cast<T,U>::value,int>::type;
606  // Enabler for equality operator.
607  template <typename U>
608  using equality_enabler = typename std::enable_if<is_equality_comparable<U>::value,int>::type;
609  // Enabler for hash.
610  template <typename U>
611  using hash_enabler = typename std::enable_if<is_hashable<U>::value,int>::type;
612  // Enabler for the addition.
613  template <typename U>
614  using add_enabler = typename std::enable_if<
615  std::is_assignable<U &,decltype(std::declval<U const &>() + std::declval<U const &>()) &&>::value,int>::type;
616  // Serialization support.
617  friend class boost::serialization::access;
618  template <class Archive>
619  void save(Archive &ar, unsigned int) const
620  {
621  // First save the size.
622  const auto s = size();
623  ar & s;
624  // Save the individual elements.
625  const auto it_f = end();
626  for (auto it = begin(); it != it_f; ++it) {
627  ar & (*it);
628  }
629  }
630  template <class Archive>
631  void load(Archive &ar, unsigned int)
632  {
633  // NOTE: there are faster ways of implementing this, first of all by re-using the storage
634  // with resize(). Here we are forcing a re-allocation instead. Keep this in mind for later
635  // optimisations.
636  // Erase this.
637  *this = small_vector();

917  {
918  if (m_union.is_static()) {
919  if (size <= max_static_size) {
920  m_union.g_st().resize(static_cast<typename s_storage::size_type>(size));
921  } else {

923  // resize. The reason for putting it here as well is to ensure the safety
924  // of the static_cast below.
925  if (unlikely(resize_check_size(size,std::integral_constant<bool,need_resize_check>()))) {
926  piranha_throw(std::bad_alloc,);
927  }
928  // Move the existing elements into new dynamic storage.
929  const auto d_size = static_cast<typename d_storage::size_type>(size);
930  d_storage tmp_d;
931  tmp_d.reserve(d_size);
932  // NOTE: this will not throw, as tmp_d is guaranteed to be of adequate size
933  // and thus push_back() will not fail.
934  std::move(m_union.g_st().begin(),m_union.g_st().end(),std::back_inserter(tmp_d));
935  // Fill in the missing elements.
936  tmp_d.resize(d_size);
937  // Destroy static, move in dynamic.
938  m_union.g_st().~s_storage();

943  piranha_throw(std::bad_alloc,);

1027  struct adder

1035  // promoted to int during addition, hence resulting in a warning when casting back down

1037  template <typename U>
1038  struct adder<U,typename std::enable_if<std::is_integral<U>::value>::type>

1045  template <typename U>
1046  void push_back_impl(U &&x)
1047  {
1048  if (m_union.is_static()) {
1049  if (m_union.g_st().size() == m_union.g_st().max_size) {
1050  // Create a new dynamic vector, and move in the current
1051  // elements from static storage.
1052  using d_size_type = typename d_storage::size_type;
1053  // NOTE: this check ensures that the d_storage::max_size is strictly greater than
1054  // the current (static) size (equal to max static size). This means we can compute safely
1055  // current size + 1 while casting to dynamic storage size type and try to allocate enough space for
1056  // the std::move and push_back() below.
1057  // NOTE: this check is analogous to current_size + 1 > d_storage::max_size, i.e., the same
1058  // check in resize(), but it will use static const variables written as this.
1059  if (unlikely(s_storage::max_size >= d_storage::max_size)) {
1060  piranha_throw(std::bad_alloc,);
1061  }
1062  d_storage tmp_d;
1063  tmp_d.reserve(static_cast<d_size_type>(static_cast<d_size_type>(m_union.g_st().max_size) + 1u));
1064  std::move(m_union.g_st().begin(),m_union.g_st().end(),std::back_inserter(tmp_d));
1065  // Push back the new element.
1066  tmp_d.push_back(std::forward<U>(x));
1067  // Now destroy the current static storage, and move-construct new dynamic storage.
1068  // NOTE: in face of custom allocators here we should be ok, as move construction
1069  // of custom alloc will not throw and it will preserve the ownership of the moved-in elements.
1070  m_union.g_st().~s_storage();
1071  ::new (static_cast<void *>(&(m_union.m_dy))) d_storage(std::move(tmp_d));
1072  } else {
1073  m_union.g_st().push_back(std::forward<U>(x));
1074  }
1075  } else {
1076  // In case we are already in dynamic storage, don't do anything special.
1077  m_union.g_dy().push_back(std::forward<U>(x));
1078  }
1079  }
