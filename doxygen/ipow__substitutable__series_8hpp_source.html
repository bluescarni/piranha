90  static const unsigned value = static_cast<unsigned>(has_ipow_subs<typename Term::cf_type,T>::value) +
91  (static_cast<unsigned>(key_has_ipow_subs<typename Term::key_type,T>::value) << 1u);

94  template <typename T, typename Term>
95  using cf_subs_type = decltype(math::ipow_subs(std::declval<typename Term::cf_type const &>(),std::declval<std::string const &>(),
96  std::declval<const integer &>(),std::declval<T const &>()));
97  template <typename T, typename Term>
98  using ret_type_1 = decltype(std::declval<const cf_subs_type<T,Term> &>() * std::declval<Derived const &>());
99  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 1u,int>::type = 0>
100  static ret_type_1<T,Term> subs_term_impl(const Term &t, const std::string &name, const integer &n, const T &x, const symbol_set &s_set)
101  {
102  Derived tmp;
103  tmp.set_symbol_set(s_set);
104  tmp.insert(Term(typename Term::cf_type(1),t.m_key));
105  // NOTE: use moves here in case the multiplication can take advantage.
106  return math::ipow_subs(t.m_cf,name,n,x) * std::move(tmp);
107  }
108  // Case 2: subs only on key.
109  template <typename T, typename Term>
110  using k_subs_type = typename decltype(std::declval<const typename Term::key_type &>().ipow_subs(std::declval<const std::string &>(),
111  std::declval<const integer &>(),std::declval<const T &>(),std::declval<const symbol_set &>()))::value_type::first_type;
112  template <typename T, typename Term>
113  using ret_type_2_ = decltype(std::declval<Derived const &>() * std::declval<const k_subs_type<T,Term> &>());
114  template <typename T, typename Term>
115  using ret_type_2 = typename std::enable_if<is_addable_in_place<ret_type_2_<T,Term>>::value &&
116  std::is_constructible<ret_type_2_<T,Term>,int>::value,ret_type_2_<T,Term>>::type;
117  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 2u,int>::type = 0>
118  static ret_type_2<T,Term> subs_term_impl(const Term &t, const std::string &name, const integer &n, const T &x, const symbol_set &s_set)
119  {
120  ret_type_2<T,Term> retval(0);
121  auto ksubs = t.m_key.ipow_subs(name,n,x,s_set);
122  for (auto &p: ksubs) {
123  Derived tmp;
124  tmp.set_symbol_set(s_set);
125  tmp.insert(Term{t.m_cf,std::move(p.second)});
126  // NOTE: possible use of multadd here in the future.
127  retval += std::move(tmp) * std::move(p.first);
128  }
129  return retval;
130  }
131  // Case 3: subs on cf and key.
132  // NOTE: the checks on type 2 are already present in the alias above.
133  template <typename T, typename Term>
134  using ret_type_3 = decltype(std::declval<const cf_subs_type<T,Term> &>() * std::declval<const ret_type_2<T,Term> &>());
135  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term,T>::value == 3u,int>::type = 0>
136  static ret_type_3<T,Term> subs_term_impl(const Term &t, const std::string &name, const integer &n, const T &x, const symbol_set &s_set)
137  {
138  // Accumulator for the sum below. This is the same type resulting from case 2.
139  ret_type_2<T,Term> acc(0);
140  auto ksubs = t.m_key.ipow_subs(name,n,x,s_set);
141  auto cf_subs = math::ipow_subs(t.m_cf,name,n,x);
142  for (auto &p: ksubs) {
143  Derived tmp;
144  tmp.set_symbol_set(s_set);
145  tmp.insert(Term(typename Term::cf_type(1),std::move(p.second)));
146  // NOTE: multadd chance.
147  acc += std::move(tmp) * std::move(p.first);
148  }
149  return std::move(cf_subs) * std::move(acc);
150  }
