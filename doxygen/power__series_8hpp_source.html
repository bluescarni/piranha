166  // Total degree truncation.
167  // Case 1: coefficient can truncate, no degree or ldegree in key.
168  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,T>::value &&
169  (ps_term_score<Term>::value >> 1u) == 0u,int>::type = 0>
170  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &)
171  {
172  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree),t.m_key));
173  }
174  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.
175  // NOTE: here we do not have support for key truncation (yet), so we decide based on the low degree of the key:
176  // if it is larger than the max degree, remove the term, otherwise keep it - it is an all-or-nothing scenario.
177  template <typename Term, typename T, typename std::enable_if<!has_truncate_degree<typename Term::cf_type,T>::value &&
178  (ps_term_score<Term>::value >> 1u) == 1u &&
179  is_greater_than_comparable<decltype(std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set &>())),T>::value,int>::type = 0>
180  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
181  {
182  if (t.m_key.ldegree(s) > max_degree) {
183  // Term must be discarded.
184  return std::make_pair(false,Term());
185  } else {
186  // Keep the term as it is.
187  return std::make_pair(true,Term(t.m_cf,t.m_key));
188  }
189  }
190  // Case 3: coefficient can truncate, degree and ldegree in key, the new max degree type can be used in the coefficient truncation.
191  // NOTE: again, no key truncation, thus we decrement the real degree of the coefficient by the low degree of the key. This way we will keep
192  // all the important parts, plus some garbage.
193  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,
194  decltype(std::declval<const T &>() - std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set &>()))>::value &&
195  (ps_term_score<Term>::value >> 1u) == 1u,int>::type = 0>
196  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
197  {
198  // The truncation level for the coefficient must be modified in order to take
199  // into account the degree of the key.
200  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree - t.m_key.ldegree(s)),t.m_key));
201  }

482  static void set_auto_truncate_degree(const U &max_degree)
483  {
484  // Init out for exception safety.
485  auto new_degree(safe_cast<degree_type<T>>(max_degree));
486  // Initialisation of function-level statics is thread-safe, no need to lock. We get
487  // a ref here because the initialisation of the static could throw in principle,

489  auto &at_dm = get_at_degree_max();

491  s_at_degree_mode = 1;
492  at_dm = std::move(new_degree);
493  // This should not throw (a vector of strings, destructors and deallocation should be noexcept).
494  s_at_degree_names.clear();
495  }

584  template <typename T = power_series, at_degree_enabler<T> = 0>
585  void auto_truncate()
586  {
587  // This are local variables into which the global ones will be copied.
588  int at_degree_mode;
589  std::vector<std::string> names;
590  degree_type<T> max_degree;
591  {
592  std::lock_guard<std::mutex> lock(s_at_degree_mutex);
593  if (!s_at_degree_mode) {
594  // Nothing to do if no auto truncation is requested.
595  return;
596  }
597  // Acquire the global vars into local. These are all read operations,
598  // no problems if they throw.
599  max_degree = get_at_degree_max();
600  names = s_at_degree_names;
601  at_degree_mode = s_at_degree_mode;
602  }
603  // No more locking needed from here.
604  switch (at_degree_mode) {
605  case 1:
606  *static_cast<Derived *>(this) = truncate_degree(max_degree);
607  break;
608  case 2:
609  *static_cast<Derived *>(this) = truncate_degree(max_degree,names);
610  }
611  // Do the auto truncation from the parent.
612  parent_auto_truncate();
613  }
614  private:
615  // Static data for auto_truncate_degree.
616  static std::mutex s_at_degree_mutex;
617  static int s_at_degree_mode;
618  static std::vector<std::string> s_at_degree_names;
