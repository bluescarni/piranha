811 template <typename T>
812 inline detail::pbracket_type<T> pbracket(const T &f, const T &g, const std::vector<std::string> &p_list,
813  const std::vector<std::string> &q_list)
814 {
815  using return_type = detail::pbracket_type<T>;
816  if (p_list.size() != q_list.size()) {
817  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
818  }
819  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
820  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
821  }
822  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
823  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
824  }
825  return_type retval = return_type(0);
826  for (decltype(p_list.size()) i = 0u; i < p_list.size(); ++i) {
827  // NOTE: could use multadd/sub here, if we implement it for series.
828  retval = retval + partial(f,q_list[i]) * partial(g,p_list[i]);
829  retval = retval - partial(f,p_list[i]) * partial(g,q_list[i]);
830  }
831  return retval;
832 }

865 template <typename T>
866 inline bool is_canonical_impl(const std::vector<T const *> &new_p, const std::vector<T const *> &new_q,
867  const std::vector<std::string> &p_list, const std::vector<std::string> &q_list)
868 {
869  using p_type = decltype(math::pbracket(*new_q[0],*new_p[0],p_list,q_list));
870  if (p_list.size() != q_list.size()) {
871  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
872  }
873  if (new_p.size() != new_q.size()) {
874  piranha_throw(std::invalid_argument,"the number of new coordinates is different from the number of new momenta");
875  }
876  if (p_list.size() != new_p.size()) {
877  piranha_throw(std::invalid_argument,"the number of new momenta is different from the number of momenta");
878  }
879  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
880  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
881  }
882  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
883  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
884  }
885  const auto size = new_p.size();
886  for (decltype(new_p.size()) i = 0u; i < size; ++i) {
887  for (decltype(new_p.size()) j = 0u; j < size; ++j) {
888  // NOTE: no need for actually doing computations when i == j.
889  if (i != j && !math::is_zero(math::pbracket(*new_p[i],*new_p[j],p_list,q_list))) {
890  return false;
891  }
892  if (i != j && !math::is_zero(math::pbracket(*new_q[i],*new_q[j],p_list,q_list))) {
893  return false;
894  }
895  // Poisson bracket needs to be zero for i != j, one for i == j.
896  // NOTE: cast from bool to int is always 0 or 1.
897  if (math::pbracket(*new_q[i],*new_p[j],p_list,q_list) != p_type(static_cast<int>(i == j))) {
898  return false;
899  }
900  }
901  }
902  return true;
903 }
