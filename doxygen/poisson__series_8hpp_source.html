180  // Convert an input polynomial to a Poisson series of type RetT. The conversion
181  // will be successful if the polynomial can be reduced to an integral linear combination of
182  // symbols.
183  template <bool IsCos, typename RetT, typename Poly>
184  static RetT poly_to_ps(const Poly &poly)
185  {
186  // Shortcuts.
187  typedef typename RetT::term_type term_type;
188  typedef typename term_type::cf_type cf_type;
189  typedef typename term_type::key_type key_type;
190  typedef typename key_type::value_type value_type;
191  // Try to get the integral combination from the poly coefficient.
192  auto lc = poly.integral_combination();
193  // Change sign if needed.
194  bool sign_change = false;
195  if (!lc.empty() && lc.begin()->second.sign() < 0) {
196  std::for_each(lc.begin(),lc.end(),[](std::pair<const std::string,integer> &p) {p.second.negate();});
197  sign_change = true;
198  }
199  // Return value.
200  RetT retval;
201  // Build vector of integral multipliers.
202  std::vector<value_type> v;
203  for (auto it = lc.begin(); it != lc.end(); ++it) {
204  retval.m_symbol_set.add(it->first);
205  // NOTE: this should probably be a safe_cast.
206  // The value type here could be anything, and not guaranteed to be castable,
207  // even if in the current implementation this is guaranteed to be a signed
208  // int of some kind.
209  v.push_back(static_cast<value_type>(it->second));
210  }
211  // Build term, fix signs and flavour and move-insert it.
212  term_type term(cf_type(1),key_type(v.begin(),v.end()));
213  if (!IsCos) {
214  term.m_key.set_flavour(false);
215  if (sign_change) {
216  // NOTE: negate is supported by any coefficient type.
217  math::negate(term.m_cf);
218  }
219  }
220  retval.insert(std::move(term));
221  return retval;
222  }

368  template <typename T = poisson_series>
369  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
370  const std::true_type &) const
371  {
372  typedef typename base::term_type term_type;
373  typedef typename term_type::cf_type cf_type;
374  integer degree;
375  try {
376  degree = safe_cast<integer>(math::degree(term.m_cf,{s.get_name()}));
377  } catch (const std::invalid_argument &) {
378  piranha_throw(std::invalid_argument,
379  "unable to perform Poisson series integration: cannot convert polynomial degree to an integer");
380  }
381  // If the variable is in both cf and key, and the cf degree is negative, we cannot integrate.
382  if (degree.sign() < 0) {
383  piranha_throw(std::invalid_argument,
384  "unable to perform Poisson series integration: polynomial coefficient has negative integral degree");
385  }
386  // Init retval and auxiliary quantities for the iteration.
387  auto key_int = term.m_key.integrate(s,this->m_symbol_set);
388  // NOTE: here we are sure that the variable is contained in the monomial.
389  piranha_assert(key_int.first != 0);
390  poisson_series tmp;
391  tmp.set_symbol_set(this->m_symbol_set);
392  // NOTE: not move for .second, as it is needed in the loop below.
393  tmp.insert(term_type(cf_type(1),key_int.second));
394  i_cf_type_p<T> p_cf(term.m_cf / std::move(key_int.first));
395  integrate_type<T> retval(p_cf * tmp);
396  for (integer i(1); i <= degree; ++i) {
397  key_int = key_int.second.integrate(s,this->m_symbol_set);
398  piranha_assert(key_int.first != 0);
399  p_cf = math::partial(p_cf / std::move(key_int.first),s.get_name());
400  // Sign change due to the second portion of integration by part.
401  math::negate(p_cf);
402  tmp = poisson_series{};
403  tmp.set_symbol_set(this->m_symbol_set);
404  // NOTE: don't move second.
405  tmp.insert(term_type(cf_type(1),key_int.second));
406  retval += p_cf * tmp;
407  }
408  return retval;
409  }

418  // Time integration.
419  // First the implementation for divisor series.
420  template <typename T>
421  using ti_type_divisor_ = decltype((std::declval<const T &>() * std::declval<const typename T::term_type::cf_type &>()) /
422  std::declval<const integer &>());
423  template <typename T>
424  using ti_type_divisor = typename std::enable_if<std::is_constructible<ti_type_divisor_<T>,int>::value &&
425  is_addable_in_place<ti_type_divisor_<T>>::value &&
426  std::is_base_of<detail::divisor_series_tag,typename T::term_type::cf_type>::value,ti_type_divisor_<T>>::type;
427  template <typename T = poisson_series>
428  ti_type_divisor<T> t_integrate_impl(const std::vector<std::string> &names) const
429  {
430  using return_type = ti_type<T>;
431  // Calling series types.
432  using term_type = typename base::term_type;
433  // The value type of the trigonometric key.
434  using k_value_type = typename term_type::key_type::value_type;
435  // Divisor series types.
436  using d_series_type = typename base::term_type::cf_type;
437  using d_term_type = typename d_series_type::term_type;
438  using d_cf_type = typename d_term_type::cf_type;
439  using d_key_type = typename d_term_type::key_type;
440  // Initialise the return value.
441  return_type retval(0);
442  // Setup of the symbol set.
443  piranha_assert(names.size() == this->m_symbol_set.size());
444  symbol_set div_symbols;
445  for (const auto &name: names) {
446  div_symbols.add(name);
447  }
448  // A temp vector of integers used to normalise the divisors coming
449  // out of the integration operation from the trig keys.
450  std::vector<integer> tmp_int;
451  // Build the return value.
452  const auto it_f = this->m_container.end();
453  for (auto it = this->m_container.begin(); it != it_f; ++it) {
454  // Clear the tmp integer vector.
455  tmp_int.clear();
456  // Get the vector of trigonometric multipliers.
457  const auto trig_vector = it->m_key.unpack(this->m_symbol_set);
458  // Copy it over to the tmp_int as integer values.
459  std::transform(trig_vector.begin(),trig_vector.end(),std::back_inserter(tmp_int),
460  [](const k_value_type &n) {return integer(n);});
461  // Determine the common divisor.
462  // NOTE: both the divisor and the trigonometric key share the canonical form in which the
463  // first nonzero multiplier is positive, so we don't need to account for sign flips when
464  // constructing a divisor from the trigonometric part. We just need to take care
465  // of the common divisor.
466  integer cd(0);
467  bool first_nonzero_found = false;
468  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
469  // NOTE: gcd is safe, operating on integers.
470  cd = math::gcd(cd,*it2);
471  if (!first_nonzero_found && !math::is_zero(*it2)) {
472  piranha_assert(*it2 > 0);
473  first_nonzero_found = true;
474  }
475  }
476  if (unlikely(math::is_zero(cd))) {
477  piranha_throw(std::invalid_argument,"an invalid trigonometric term was encountered while "
478  "attempting a time integration");
479  }
480  // Take the abs of the cd.
481  cd = cd.abs();
482  // Divide the vector by the common divisor.
483  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
484  *it2 /= cd;
485  }
486  // Build the temporary divisor series from the trigonometric arguments.
487  d_series_type div_series;
488  div_series.set_symbol_set(div_symbols);
489  // Build the divisor key.
490  typename d_key_type::value_type exponent(1);
491  d_key_type div_key;
492  div_key.insert(tmp_int.begin(),tmp_int.end(),exponent);
493  // Finish building the temporary divisor series.
494  div_series.insert(d_term_type(d_cf_type(1),std::move(div_key)));
495  // Temporary Poisson series from the current term, with the trig flavour flipped.
496  poisson_series tmp_ps;
497  tmp_ps.set_symbol_set(this->m_symbol_set);
498  auto tmp_key = it->m_key;
499  tmp_key.set_flavour(!tmp_key.get_flavour());
500  tmp_ps.insert(term_type(it->m_cf,std::move(tmp_key)));
501  // Update the return value.
502  auto tmp = (std::move(tmp_ps) * std::move(div_series)) / cd;
503  // It also needs a negation, if the original trig key is a sine.
504  if (!it->m_key.get_flavour()) {
505  math::negate(tmp);
506  }
507  retval += std::move(tmp);
508  }
509  return retval;
510  }
511  // Implementation when the coefficient is a rational function.
512  template <typename T = poisson_series, typename std::enable_if<std::is_base_of<
513  detail::rational_function_tag,typename T::term_type::cf_type>::value,int>::type = 0>
514  T t_integrate_impl(const std::vector<std::string> &names) const
515  {
516  // Types from base.
517  using term_type = typename base::term_type;
518  using r_type = typename term_type::cf_type;
519  using p_type = typename r_type::p_type;
520  using key_type = typename term_type::key_type;
521  piranha_assert(names.size() == this->get_symbol_set().size());
522  T retval;
523  retval.set_symbol_set(this->get_symbol_set());
524  for (const auto &t: this->_container()) {
525  // Get the flavour of the current trig monomial.
526  const bool flavour = t.m_key.get_flavour();
527  // Get the vector of trigonometric multipliers.
528  const auto trig_vector = t.m_key.unpack(this->m_symbol_set);
529  // Construct a polynomial from the extracted multipliers, using
530  // the names in linear combination.
531  p_type tmp;
532  for (decltype(trig_vector.size()) i = 0u; i < trig_vector.size(); ++i) {
533  tmp += p_type{names[static_cast<decltype(names.size())>(i)]} * trig_vector[i];
534  }
535  if (math::is_zero(tmp)) {
536  piranha_throw(std::invalid_argument,"an invalid trigonometric term was encountered while "
537  "attempting a time integration");
538  }
539  // Construct the new coefficient from the current coefficient,
540  // divided by the newly constructed poly.
541  r_type r{t.m_cf / std::move(tmp)};
542  // Need to negate the coefficient if the current trig monomial is a sine.
543  if (!flavour) {
544  math::negate(r);
545  }
546  // Create a copy of the current trig monomial and flip its flavour.
547  key_type tmp_key{t.m_key};
548  tmp_key.set_flavour(!flavour);
549  // Create new term from new coefficient and key, and insert it.
550  retval.insert(term_type{std::move(r),std::move(tmp_key)});
551  }
552  return retval;
553  }

674  template <typename T = poisson_series>
675  integrate_type<T> integrate(const std::string &name) const
676  {
677  typedef typename base::term_type term_type;
678  typedef typename term_type::cf_type cf_type;
679  // Turn name into symbol.
680  const symbol s(name);
681  // Init the return value.
682  integrate_type<T> retval(0);
683  const auto it_f = this->m_container.end();
684  for (auto it = this->m_container.begin(); it != it_f; ++it) {
685  // Integrate the key first.
686  auto key_int = it->m_key.integrate(s,this->m_symbol_set);
687  // If the variable does not appear in the monomial, try deferring the integration
688  // to the coefficient.
689  if (key_int.first == 0) {
690  integrate_coefficient_only(retval,it,name);
691  continue;
692  }
693  // The variable is in the monomial, let's check if the variable is also in the coefficient.
694  if (math::is_zero(math::partial(it->m_cf,name))) {
695  // No variable in the coefficient, proceed with the integrated key and divide by multiplier.
696  poisson_series tmp;
697  tmp.set_symbol_set(this->m_symbol_set);
698  tmp.insert(term_type(cf_type(1),std::move(key_int.second)));
699  retval += (std::move(tmp) * it->m_cf) / key_int.first;
700  } else {
701  // With the variable both in the coefficient and the key, we only know how to proceed with polynomials.
702  retval += integrate_impl(s,*it,std::integral_constant<bool,std::is_base_of<detail::polynomial_tag,Cf>::value>());
703  }
704  }
705  return retval;

800  template <typename T>
801  using call_enabler = typename std::enable_if<key_is_multipliable<typename T::term_type::cf_type,
802  typename T::term_type::key_type>::value,int>::type;
803  void divide_by_two(Series &s) const
804  {
805  // NOTE: if we ever implement multi-threaded series division we most likely need
806  // to revisit this.
807  piranha_assert(this->m_n_threads > 0u);
808  if (this->m_n_threads == 1u) {
809  // This is possible, as the requirements of series divisibility and trig key
810  // multipliability overlap.
811  s /= 2;
812  } else {
813  using bucket_size_type = typename base::bucket_size_type;
814  using term_type = typename Series::term_type;
815  auto &container = s._container();
816  std::atomic<bucket_size_type> total_erase_count(0u);
817  auto divider = [&container,&total_erase_count,this](bucket_size_type start_idx, bucket_size_type end_idx) {
818  // A vector of terms to be erased at each bucket iteration.
819  std::vector<term_type> term_list;
820  // Total number of terms erased by this thread.
821  bucket_size_type erase_count = 0u;
822  for (; start_idx != end_idx; ++start_idx) {
823  // Reset the list of terms to be erased.
824  term_list.clear();
825  const auto &list = container._get_bucket_list(start_idx);
826  for (const auto &t: list) {
827  t.m_cf /= 2;
828  if (unlikely(t.is_ignorable(this->m_ss))) {
829  term_list.push_back(t);
830  }
831  }
832  for (const auto &t: term_list) {
833  container._erase(container._find(t,start_idx));
834  erase_count = static_cast<bucket_size_type>(erase_count + 1u);
835  }
836  }
837  // Update the global counter of erased terms.
838  total_erase_count += erase_count;
839  };
840  // Buckets per thread.
841  const auto bpt = static_cast<bucket_size_type>(container.bucket_count() / this->m_n_threads);
842  // Go with the threads.
843  future_list<decltype(thread_pool::enqueue(0u,divider,0u,0u))> ff_list;
844  try {
845  for (unsigned i = 0u; i < this->m_n_threads; ++i) {
846  const auto start_idx = static_cast<bucket_size_type>(bpt * i);
847  // Special casing for the last thread.
848  const auto end_idx = (i == this->m_n_threads - 1u) ? container.bucket_count() :
849  static_cast<bucket_size_type>(bpt * (i + 1u));
850  ff_list.push_back(thread_pool::enqueue(i,divider,start_idx,end_idx));
851  }
852  // First let's wait for everything to finish.
853  ff_list.wait_all();
854  // Then, let's handle the exceptions.
855  ff_list.get_all();
856  } catch (...) {
857  ff_list.wait_all();
858  // Clear out the container as it might be in an inconsistent state.
859  container.clear();
860  throw;
861  }
862  // Final size update - all of this is noexcept.
863  const auto tot = total_erase_count.load();
864  piranha_assert(tot <= s.size());
865  container._update_size(static_cast<bucket_size_type>(s.size() - tot));
866  }
867  }

870  using base::base;

887  divide_by_two(retval);
