257  template <typename T = polynomial>
258  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
259  const std::true_type &) const
260  {
261  typedef typename base::term_type term_type;
262  typedef typename term_type::cf_type cf_type;
263  typedef typename term_type::key_type key_type;
264  // Get the partial degree of the monomial in integral form.
265  integer degree;
266  const symbol_set::positions pos(this->m_symbol_set,symbol_set{s});
267  try {
268  degree = safe_cast<integer>(term.m_key.degree(pos,this->m_symbol_set));
269  } catch (const std::invalid_argument &) {
270  piranha_throw(std::invalid_argument,
271  "unable to perform polynomial integration: cannot extract the integral form of an exponent");
272  }
273  // If the degree is negative, integration by parts won't terminate.
274  if (degree.sign() < 0) {
275  piranha_throw(std::invalid_argument,
276  "unable to perform polynomial integration: negative integral exponent");
277  }
278  polynomial tmp;
279  tmp.set_symbol_set(this->m_symbol_set);
280  key_type tmp_key = term.m_key;
281  tmp.insert(term_type(cf_type(1),tmp_key));
282  i_cf_type_p<T> i_cf(math::integrate(term.m_cf,s.get_name()));
283  integrate_type<T> retval(i_cf * tmp);
284  for (integer i(1); i <= degree; ++i) {
285  // Update coefficient and key. These variables are persistent across loop iterations.
286  auto partial_key = tmp_key.partial(pos,this->m_symbol_set);
287  i_cf = math::integrate(i_cf,s.get_name()) * std::move(partial_key.first);
288  // Account for (-1)**i.
289  math::negate(i_cf);
290  // Build the other factor from the derivative of the monomial.
291  tmp = polynomial{};
292  tmp.set_symbol_set(this->m_symbol_set);
293  tmp_key = std::move(partial_key.second);
294  // NOTE: don't move tmp_key, as it needs to hold a valid value
295  // for the next loop iteration.
296  tmp.insert(term_type(cf_type(1),tmp_key));
297  retval += i_cf * tmp;
298  }
299  return retval;
300  }

477  integrate_type<T> integrate(const std::string &name) const
478  {
479  typedef typename base::term_type term_type;
480  typedef typename term_type::cf_type cf_type;
481  // Turn name into symbol.
482  const symbol s(name);
483  integrate_type<T> retval(0);
484  const auto it_f = this->m_container.end();
485  for (auto it = this->m_container.begin(); it != it_f; ++it) {
486  // If the derivative of the coefficient is null, we just need to deal with
487  // the integration of the key.
488  if (math::is_zero(math::partial(it->m_cf,name))) {
489  polynomial tmp;
490  symbol_set sset = this->m_symbol_set;
491  // If the variable does not exist in the arguments set, add it.
492  if (!std::binary_search(sset.begin(),sset.end(),s)) {
493  sset.add(s);
494  }
495  tmp.set_symbol_set(sset);
496  auto key_int = it->m_key.integrate(s,this->m_symbol_set);
497  tmp.insert(term_type(cf_type(1),std::move(key_int.second)));
498  retval += (tmp * it->m_cf) / key_int.first;
499  } else {
500  retval += integrate_impl(s,*it,std::integral_constant<bool,is_integrable<cf_type>::value>());
501  }
502  }

570  explicit series_multiplier(const Series &s1, const Series &s2):base(s1,s2)
571  {
572  if (unlikely(this->m_s1->empty() || this->m_s2->empty())) {
573  return;
574  }
575  // NOTE: here we are sure about this since the symbol set in a series should never
576  // overflow the size of the limits, as the check for compatibility in Kronecker monomial
577  // would kick in.
578  piranha_assert(this->m_s1->m_symbol_set.size() < ka::get_limits().size());
579  piranha_assert(this->m_s1->m_symbol_set == this->m_s2->m_symbol_set);
580  const auto &limits = ka::get_limits()[this->m_s1->m_symbol_set.size()];
581  // NOTE: We need to check that the exponents of the monomials in the result do not
582  // go outside the bounds of the Kronecker codification. We need to unpack all monomials
583  // in the operands and examine them, we cannot operate on the codes for this.
584  auto it1 = this->m_v1.begin();
585  auto it2 = this->m_v2.begin();
586  // Initialise minmax values.
587  std::vector<std::pair<value_type,value_type>> minmax_values1;
588  std::vector<std::pair<value_type,value_type>> minmax_values2;
589  auto tmp_vec1 = (*it1)->m_key.unpack(this->m_s1->m_symbol_set);
590  auto tmp_vec2 = (*it2)->m_key.unpack(this->m_s1->m_symbol_set);
591  // Bounds of the Kronecker representation for each component.
592  const auto &minmax_vec = std::get<0u>(limits);
593  std::transform(tmp_vec1.begin(),tmp_vec1.end(),std::back_inserter(minmax_values1),[](const value_type &v) {
594  return std::make_pair(v,v);
595  });
596  std::transform(tmp_vec2.begin(),tmp_vec2.end(),std::back_inserter(minmax_values2),[](const value_type &v) {
597  return std::make_pair(v,v);
598  });
599  // Find the minmaxs.
600  for (; it1 != this->m_v1.end(); ++it1) {
601  tmp_vec1 = (*it1)->m_key.unpack(this->m_s1->m_symbol_set);
602  piranha_assert(tmp_vec1.size() == minmax_values1.size());
603  std::transform(minmax_values1.begin(),minmax_values1.end(),tmp_vec1.begin(),minmax_values1.begin(),
604  [](const std::pair<value_type,value_type> &p, const value_type &v) {
605  return std::make_pair(
606  v < p.first ? v : p.first,
607  v > p.second ? v : p.second
608  );
609  });
610  }
611  for (; it2 != this->m_v2.end(); ++it2) {
612  tmp_vec2 = (*it2)->m_key.unpack(this->m_s2->m_symbol_set);
613  piranha_assert(tmp_vec2.size() == minmax_values2.size());
614  std::transform(minmax_values2.begin(),minmax_values2.end(),tmp_vec2.begin(),minmax_values2.begin(),
615  [](const std::pair<value_type,value_type> &p, const value_type &v) {
616  return std::make_pair(
617  v < p.first ? v : p.first,
618  v > p.second ? v : p.second
619  );
620  });
621  }
622  // Compute the sum of the two minmaxs, using multiprecision to avoid overflow.
623  // NOTE: first store in m_minmax_values the ranges of the result only, below we will update this with the range
624  // of the operands.
625  std::transform(minmax_values1.begin(),minmax_values1.end(),minmax_values2.begin(),
626  std::back_inserter(m_minmax_values),[](const std::pair<value_type,value_type> &p1,
627  const std::pair<value_type,value_type> &p2) {
628  return std::make_pair(integer(p1.first) + integer(p2.first),integer(p1.second) + integer(p2.second));
629  });
630  piranha_assert(m_minmax_values.size() == minmax_vec.size());
631  piranha_assert(m_minmax_values.size() == minmax_values1.size());
632  piranha_assert(m_minmax_values.size() == minmax_values2.size());
633  for (decltype(m_minmax_values.size()) i = 0u; i < m_minmax_values.size(); ++i) {
634  if (unlikely(m_minmax_values[i].first < -minmax_vec[i] || m_minmax_values[i].second > minmax_vec[i])) {
635  piranha_throw(std::overflow_error,"Kronecker monomial components are out of bounds");
636  }
637  // Update with the ranges of the operands.
638  m_minmax_values[i] = std::minmax({m_minmax_values[i].first,
639  integer(minmax_values1[i].first),integer(minmax_values2[i].first),m_minmax_values[i].second,
640  integer(minmax_values1[i].second),integer(minmax_values2[i].second)});
641  }

671  // First block: semi-open range [a,b[ of indices in first input series.
672  std::pair<index_type,index_type> m_b1;
673  // Second block (indices in second input series).
674  std::pair<index_type,index_type> m_b2;
675  // First region memory region involved in the multiplication.
676  region_type m_r1;
677  // Second memory region.
678  region_type m_r2;
679  // Boolean flag to signal if the second region is present or not.
680  bool m_second_region;
681  };
682  // Create task from indices i in first series, j in second series (semi-open intervals).
683  task_type task_from_indices(const index_type &i_start, const index_type &i_end,
684  const index_type &j_start, const index_type &j_end, const Series &retval) const
685  {
686  const auto &v1 = this->m_v1;
687  const auto &v2 = this->m_v2;
688  piranha_assert(i_start < i_end && j_start < j_end);
689  piranha_assert(i_end <= v1.size() && j_end <= v2.size());
690  const auto b_count = retval.m_container.bucket_count();
691  piranha_assert(b_count > 0u);
692  region_type r1{0u,0u}, r2{0u,0u};
693  bool second_region = false;
694  // Addition is safe because of the limits on the max bucket count of hash_set.
695  const auto a = retval.m_container._bucket_from_hash(v1[i_start]->hash()) +
696  retval.m_container._bucket_from_hash(v2[j_start]->hash()),
697  // NOTE: we are sure that the tasks are not empty, so the -1 is safe.
698  b = retval.m_container._bucket_from_hash(v1[i_end - 1u]->hash()) +
699  retval.m_container._bucket_from_hash(v2[j_end - 1u]->hash());
700  // NOTE: using <= here as [a,b] is now a closed interval.
701  piranha_assert(a <= b);
702  piranha_assert(b <= b_count * 2u - 2u);
703  if (b < b_count || a >= b_count) {
704  r1.first = a % b_count;
705  r1.second = b % b_count;
706  } else {
707  second_region = true;
708  r1.first = a;
709  r1.second = b_count - 1u;
710  r2.first = 0u;
711  r2.second = b % b_count;
712  }
713  // Correct the case in which a second region reaches the first one (thus
714  // covering the whole range of buckets).
715  if (second_region && r2.second >= r1.first) {
716  second_region = false;
717  r1.first = 0u;
718  r1.second = b_count - 1u;
719  r2.first = 0u;
720  r2.second = 0u;
721  }
722  piranha_assert(r1.first <= r1.second);
723  piranha_assert(!second_region || (r2.first <= r2.second && r2.second < r1.first));
724  return task_type{std::make_pair(i_start,i_end),std::make_pair(j_start,j_end),r1,r2,second_region};
725  }
726  // Functor to check if region r does not overlap any of the busy ones.
727  template <typename RegionSet>
728  static bool region_checker(const region_type &r, const RegionSet &busy_regions)
729  {
730  if (busy_regions.empty()) {
731  return true;
732  }
733  // NOTE: lower bound means that it_b->first >= r.first.
734  auto it_b = busy_regions.lower_bound(r);
735  // Handle end().
736  if (it_b == busy_regions.end()) {
737  // NOTE: safe because busy_regions is not empty.
738  --it_b;
739  // Now check that the region in it_b does not overlap with
740  // the current one.
741  piranha_assert(it_b->first < r.first);
742  return it_b->second < r.first;
743  }
744  if (r.second >= it_b->first) {
745  // The end point of r1 overlaps the start point of *it_b, no good.
746  return false;
747  }
748  // Lastly, we have to check that the previous element (if any) does not
749  // overlap r.first.
750  if (it_b != busy_regions.begin()) {
751  --it_b;
752  if (it_b->second >= r.first) {
753  return false;
754  }
755  }
756  return true;
757  }
758  // Check that a set of busy regions is well-formed, for debug purposes.
759  template <typename RegionSet>
760  static bool region_set_checker(const RegionSet &busy_regions)
761  {
762  if (busy_regions.empty()) {
763  return true;
764  }
765  const auto it_f = busy_regions.end();
766  auto it = busy_regions.begin();
767  auto prev_it(it);
768  ++it;
769  for (; it != it_f; ++it, ++prev_it) {
770  if (it->first > it->second || prev_it->first > prev_it->second) {
771  return false;
772  }
773  if (prev_it->second >= it->first) {
774  return false;
775  }
776  }
777  return true;
778  }
779  Series execute() const
780  {
781  const index_type size1 = this->m_v1.size(), size2 = this->m_v2.size();
782  // Do not do anything if one of the two series is empty, just return an empty series.
783  if (unlikely(!size1 || !size2)) {
784  return Series{};
785  }
786  // This check is done here to avoid controlling the number of elements of the output series
787  // at every iteration of the functor.
788  const auto max_size = integer(size1) * size2;
789  if (unlikely(max_size > std::numeric_limits<bucket_size_type>::max())) {
790  piranha_throw(std::overflow_error,"possible overflow in series size");
791  }
792  // First, let's get the estimation on the size of the final series.
793  Series retval;
794  retval.m_symbol_set = this->m_s1->m_symbol_set;
795  typename Series::size_type estimate;
796  // Use the sparse functor for the estimation.
797  estimate = base::estimate_final_series_size(sparse_functor<>(&this->m_v1[0u],size1,&this->m_v2[0u],size2,retval));
798  // Correct the unlikely case of zero estimate.
799  if (unlikely(!estimate)) {
800  estimate = 1u;
801  }
802  // Get the number of threads to use.
803  const unsigned n_threads = thread_pool::use_threads(
804  integer(size1) * size2,integer(settings::get_min_work_per_thread())
805  );
806  // Rehash the retun value's container accordingly. Check the tuning flag to see if we want to use
807  // multiple threads for initing the return value.
808  // NOTE: it is important here that we use the same n_threads for multiplication and memset as
809  // we tie together pinned threads with potentially different NUMA regions.
810  const unsigned n_threads_rehash = tuning::get_parallel_memory_set() ? n_threads : 1u;
811  // NOTE: if something goes wrong here, no big deal as retval is still empty.
812  retval.m_container.rehash(boost::numeric_cast<typename Series::size_type>(std::ceil(static_cast<double>(estimate) /
813  retval.m_container.max_load_factor())),n_threads_rehash);
814  piranha_assert(retval.m_container.bucket_count());
815  // NOTE: tuning parameter.
816  if ((integer(size1) * integer(size2)) / estimate > 200) {
817  dense_multiplication(retval,n_threads);
818  } else {
819  sparse_multiplication<sparse_functor<>>(retval,n_threads);
820  }
821  // Trace the result of estimation.
822  this->trace_estimates(retval.size(),estimate);
823  return retval;
824  }
825  // Utility function to determine block sizes.
826  static std::pair<integer,integer> get_block_sizes(const index_type &size1, const index_type &size2)
827  {
828  const integer block_size(512u), job_size = block_size.pow(2u);
829  // Rescale the block sizes according to the relative sizes of the input series.
830  auto block_size1 = (block_size * size1) / size2,
831  block_size2 = (block_size * size2) / size1;
832  // Avoid having zero block sizes, or block sizes exceeding job_size.
833  if (!block_size1) {
834  block_size1 = 1u;
835  } else if (block_size1 > job_size) {
836  block_size1 = job_size;
837  }
838  if (!block_size2) {
839  block_size2 = 1u;
840  } else if (block_size2 > job_size) {
841  block_size2 = job_size;
842  }
843  return std::make_pair(std::move(block_size1),std::move(block_size2));
844  }
845  // Compare regions by lower bound.
846  struct region_comparer
847  {
848  bool operator()(const region_type &r1, const region_type &r2) const
849  {
850  return r1.first < r2.first;
851  }
852  };
853  // Function to remove the regions associated to a task from the set of busy regions.
854  template <typename RegionSet>
855  static void cleanup_regions(RegionSet &busy_regions, const task_type &task)
856  {
857  auto tmp_it = busy_regions.find(task.m_r1);
858  if (tmp_it != busy_regions.end()) {
859  busy_regions.erase(tmp_it);
860  }
861  // Deal with second region only if present.
862  if (!task.m_second_region) {
863  return;
864  }
865  tmp_it = busy_regions.find(task.m_r2);
866  if (tmp_it != busy_regions.end()) {
867  busy_regions.erase(tmp_it);
868  }
869  piranha_assert(region_set_checker(busy_regions));
870  }
871  // Dense task sorter.
872  template <typename NKType>
873  struct dts_type
874  {
875  explicit dts_type(const std::vector<NKType> &nk1, const std::vector<NKType> &nk2):
876  m_new_keys1(nk1),m_new_keys2(nk2)
877  {}
878  template <typename Task>
879  bool operator()(const Task &t1, const Task &t2) const
880  {
881  return m_new_keys1[t1.m_b1.first].first + m_new_keys2[t1.m_b2.first].first <
882  m_new_keys1[t2.m_b1.first].first + m_new_keys2[t2.m_b2.first].first;
883  }
884  const std::vector<NKType> &m_new_keys1;
885  const std::vector<NKType> &m_new_keys2;
886  };
887  // Dense multiplication method.
888  void dense_multiplication(Series &retval, const unsigned &n_threads) const
889  {
890  // Vectors of minimum / maximum values, cast to hardware int.
891  std::vector<value_type> mins;
892  std::transform(m_minmax_values.begin(),m_minmax_values.end(),std::back_inserter(mins),[](const std::pair<integer,integer> &p) {
893  return static_cast<value_type>(p.first);
894  });
895  std::vector<value_type> maxs;
896  std::transform(m_minmax_values.begin(),m_minmax_values.end(),std::back_inserter(maxs),[](const std::pair<integer,integer> &p) {
897  return static_cast<value_type>(p.second);
898  });
899  // Build the encoding vector.
900  std::vector<value_type> c_vec;
901  integer f_delta(1);
902  std::transform(m_minmax_values.begin(),m_minmax_values.end(),std::back_inserter(c_vec),
903  [&f_delta](const std::pair<integer,integer> &p) -> value_type {
904  auto old(f_delta);
905  f_delta *= p.second - p.first + 1;
906  return static_cast<value_type>(old);
907  });
908  // Try casting final delta.
909  (void)static_cast<value_type>(f_delta);
910  // Compute hmax and hmin.
911  piranha_assert(m_minmax_values.size() == c_vec.size());
912  const auto h_minmax = std::inner_product(m_minmax_values.begin(),m_minmax_values.end(),c_vec.begin(),
913  std::make_pair(integer(0),integer(0)),
914  [](const std::pair<integer,integer> &p1, const std::pair<integer,integer> &p2) {
915  return std::make_pair(p1.first + p2.first,p1.second + p2.second);
916  },[](const std::pair<integer,integer> &p, const value_type &value) {
917  return std::make_pair(p.first * value,p.second * value);
918  }
919  );
920  piranha_assert(f_delta == h_minmax.second - h_minmax.first + 1);
921  // Try casting hmax and hmin.
922  const auto hmin = static_cast<value_type>(h_minmax.first);
923  const auto hmax = static_cast<value_type>(h_minmax.second);
924  // Encoding functor.
925  typedef typename term_type::key_type::v_type unpack_type;
926  auto encoder = [&c_vec,hmin,&mins](const unpack_type &v) -> value_type {
927  piranha_assert(c_vec.size() == v.size());
928  piranha_assert(c_vec.size() == mins.size());
929  decltype(mins.size()) i = 0u;
930  value_type retval = std::inner_product(c_vec.begin(),c_vec.end(),v.begin(),value_type(0),
931  std::plus<value_type>(),[&i,&mins](const value_type &c, const value_type &n) -> value_type {
932  const decltype(mins.size()) old_i = i;
933  ++i;
934  piranha_assert(n >= mins[old_i]);
935  return static_cast<value_type>(c * (n - mins[old_i]));
936  }
937  );
938  return static_cast<value_type>(retval + hmin);
939  };
940  // Build copies of the input keys repacked according to the new Kronecker substitution. Attach
941  // also a pointer to the term.
942  typedef std::pair<value_type,term_type const *> new_key_type;
943  std::vector<new_key_type> new_keys1, new_keys2;
944  std::transform(this->m_v1.begin(),this->m_v1.end(),std::back_inserter(new_keys1),
945  [this,encoder](term_type const *ptr) {
946  return std::make_pair(encoder(ptr->m_key.unpack(this->m_s1->m_symbol_set)),ptr);
947  });
948  std::transform(this->m_v2.begin(),this->m_v2.end(),std::back_inserter(new_keys2),
949  [this,encoder](term_type const *ptr) {
950  return std::make_pair(encoder(ptr->m_key.unpack(this->m_s1->m_symbol_set)),ptr);
951  });
952  // Sort the the new keys.
953  std::stable_sort(new_keys1.begin(),new_keys1.end(),[](const new_key_type &p1, const new_key_type &p2) {
954  return p1.first < p2.first;
955  });
956  std::stable_sort(new_keys2.begin(),new_keys2.end(),[](const new_key_type &p1, const new_key_type &p2) {
957  return p1.first < p2.first;
958  });
959  // Store the sizes and compute the block sizes.
960  const index_type size1 = boost::numeric_cast<index_type>(new_keys1.size()),
961  size2 = boost::numeric_cast<index_type>(new_keys2.size());
962  piranha_assert(size1 == this->m_s1->size());
963  piranha_assert(size2 == this->m_s2->size());
964  const auto bsizes = get_block_sizes(size1,size2);
965  // Cast to hardware integers.
966  const auto bsize1 = static_cast<index_type>(bsizes.first), bsize2 = static_cast<index_type>(bsizes.second);
967  // Build the list of tasks.
968  dts_type<new_key_type> dense_task_sorter(new_keys1,new_keys2);
969  std::multiset<task_type,dts_type<new_key_type>> task_list(dense_task_sorter);
970  decltype(task_list.insert(std::declval<task_type>())) ins_result;
971  auto dense_task_from_indices = [hmin,hmax,&new_keys1,&new_keys2](const index_type &i_start, const index_type &i_end,
972  const index_type &j_start, const index_type &j_end) -> task_type
973  {
974  piranha_assert(i_end <= new_keys1.size() && j_end <= new_keys2.size());
975  piranha_assert(i_start < i_end && j_start < j_end);
976  const bucket_size_type a = boost::numeric_cast<bucket_size_type>((new_keys1[i_start].first + new_keys2[j_start].first) - hmin);
977  const bucket_size_type b = boost::numeric_cast<bucket_size_type>((new_keys1[i_end - 1u].first + new_keys2[j_end - 1u].first) - hmin);
978  piranha_assert(a <= b);
979  piranha_assert(b <= boost::numeric_cast<bucket_size_type>(hmax - hmin));
980  return task_type{std::make_pair(i_start,i_end),std::make_pair(j_start,j_end),
981  std::make_pair(a,b),std::make_pair(bucket_size_type(0u),bucket_size_type(0u)),false};
982  };
983  for (index_type i = 0u; i < size1 / bsize1; ++i) {
984  for (index_type j = 0u; j < size2 / bsize2; ++j) {
985  ins_result = task_list.insert(dense_task_from_indices(i * bsize1,(i + 1u) * bsize1,j * bsize2,(j + 1u) * bsize2));
986  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
987  }
988  if (size2 % bsize2) {
989  ins_result = task_list.insert(dense_task_from_indices(i * bsize1,(i + 1u) * bsize1,(size2 / bsize2) * bsize2,size2));
990  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
991  }
992  }
993  if (size1 % bsize1) {
994  for (index_type j = 0u; j < size2 / bsize2; ++j) {
995  ins_result = task_list.insert(dense_task_from_indices((size1 / bsize1) * bsize1,size1,j * bsize2,(j + 1u) * bsize2));
996  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
997  }
998  if (size2 % bsize2) {
999  ins_result = task_list.insert(dense_task_from_indices((size1 / bsize1) * bsize1,size1,(size2 / bsize2) * bsize2,size2));
1000  piranha_assert(ins_result->m_b1.first != ins_result->m_b1.second && ins_result->m_b2.first != ins_result->m_b2.second);
1001  }
1002  }
1003  // Prepare the storage for multiplication.
1004  // NOTE: init everything explicitly to zero, as we make no assumption about the value of a default-cted
1005  // coefficient.
1006  using cf_vector_type = std::vector<typename term_type::cf_type>;
1007  cf_vector_type cf_vector(boost::numeric_cast<typename cf_vector_type::size_type>((hmax - hmin) + 1),
1008  typename term_type::cf_type(0));
1009  if (n_threads == 1u) {
1010  // Single-thread multiplication.
1011  const auto it_f = task_list.end();
1012  for (auto it = task_list.begin(); it != it_f; ++it) {
1013  const index_type i_start = it->m_b1.first, j_start = it->m_b2.first,
1014  i_end = it->m_b1.second, j_end = it->m_b2.second;
1015  piranha_assert(i_end > i_start && j_end > j_start);
1016  for (index_type i = i_start; i < i_end; ++i) {
1017  for (index_type j = j_start; j < j_end; ++j) {
1018  const auto idx = (new_keys1[i].first + new_keys2[j].first) - hmin;
1019  piranha_assert(idx < boost::numeric_cast<value_type>(cf_vector.size()));
1020  math::multiply_accumulate(cf_vector[static_cast<decltype(cf_vector.size())>(idx)],
1021  new_keys1[i].second->m_cf,new_keys2[j].second->m_cf);
1022  }
1023  }
1024  }
1025  } else {
1026  // Set of busy bucket regions, ordered by starting point.
1027  // NOTE: we do not need a multiset, as regions that compare equal (i.e., same starting point) will not exist
1028  // at the same time in the set by design.
1029  region_comparer rc;
1030  std::set<region_type,region_comparer> busy_regions(rc);
1031  // Synchronization.
1032  std::mutex m;
1033  std::condition_variable cond;
1034  // Thread function.
1035  auto thread_function = [&cond,&m,&task_list,&busy_regions,&new_keys1,&new_keys2,hmin,&cf_vector,this] () {
1036  task_type task;
1037  while (true) {
1038  {
1039  // First, lock down everything.
1040  std::unique_lock<std::mutex> lock(m);
1041  if (task_list.empty()) {
1042  break;
1043  }
1044  // Look for a suitable task.
1045  const auto it_f = task_list.end();
1046  auto it = task_list.begin();
1047  for (; it != it_f; ++it) {
1048  piranha_assert(!it->m_second_region);
1049  // Check the region.
1050  if (this->region_checker(it->m_r1,busy_regions)) {
1051  try {
1052  // The region is ok, insert it and break the cycle.
1053  auto tmp = busy_regions.insert(it->m_r1);
1054  (void)tmp;
1055  piranha_assert(tmp.second);
1056  piranha_assert(this->region_set_checker(busy_regions));
1057  } catch (...) {
1058  // NOTE: the idea here is that in case of errors
1059  // we want to restore the original situation
1060  // as if nothing happened.
1061  this->cleanup_regions(busy_regions,*it);
1062  throw;
1063  }
1064  break;
1065  }
1066  }
1067  // We might have identified a suitable task, check it is not end().
1068  if (it == it_f) {
1069  // The thread can't do anything, will have to wait until something happens
1070  // and then re-identify a good task.
1071  cond.wait(lock);
1072  continue;
1073  }
1074  // Now we have a good task, pop it from the task list.
1075  task = *it;
1076  task_list.erase(it);
1077  }
1078  try {
1079  // Perform the multiplication on the selected task.
1080  const index_type i_start = task.m_b1.first, j_start = task.m_b2.first,
1081  i_end = task.m_b1.second, j_end = task.m_b2.second;
1082  piranha_assert(i_end > i_start && j_end > j_start);
1083  for (index_type i = i_start; i < i_end; ++i) {
1084  for (index_type j = j_start; j < j_end; ++j) {
1085  const auto idx = (new_keys1[i].first + new_keys2[j].first) - hmin;
1086  piranha_assert(idx < boost::numeric_cast<value_type>(cf_vector.size()));
1087  math::multiply_accumulate(cf_vector[static_cast<decltype(cf_vector.size())>(idx)],
1088  new_keys1[i].second->m_cf,new_keys2[j].second->m_cf);
1089  }
1090  }
1091  } catch (...) {
1092  // Re-acquire the lock.
1093  std::lock_guard<std::mutex> lock(m);
1094  // Cleanup the regions.
1095  this->cleanup_regions(busy_regions,task);
1096  // Notify all waiting threads that a region was removed from the busy set.
1097  cond.notify_all();
1098  throw;
1099  }
1100  {
1101  // Re-acquire the lock.
1102  std::lock_guard<std::mutex> lock(m);
1103  // Take out the regions in which we just wrote from the set of busy regions.
1104  this->cleanup_regions(busy_regions,task);
1105  // Notify all waiting threads that a region was removed from the busy set.
1106  cond.notify_all();
1107  }
1108  }
1109  };
1110  // NOTE: one of the fundamental requirements here is that we wait for pending
1111  // tasks, in case of errors, before getting out of this function: thread_function
1112  // contains references to local variables, if we get out of here before the tasks are finished
1113  // memory corruption will occur.
1114  future_list<decltype(thread_pool::enqueue(0u,thread_function))> f_list;
1115  try {
1116  for (unsigned i = 0u; i < n_threads; ++i) {
1117  // NOTE: enqueue() will either happen or it won't, we only care
1118  // about memory allocation errors in push_back() here. In such case,
1119  // push_back() will wait on the temporary future from enqueue
1120  // before returning the exception.
1121  f_list.push_back(thread_pool::enqueue(i,thread_function));
1122  }
1123  // First let's wait for everything to finish.
1124  f_list.wait_all();
1125  // Then, let's handle the exceptions.
1126  f_list.get_all();
1127  } catch (...) {
