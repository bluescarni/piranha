54 // TMP to determine size type big enough to represent Size. The candidate types are the fundamental unsigned integer types.
55 using static_vector_size_types = std::tuple<unsigned char, unsigned short, unsigned, unsigned long, unsigned long long>;

57 template <std::size_t Size, std::size_t Index = 0u>
58 struct static_vector_size_type
59 {
60  using candidate_type = typename std::tuple_element<Index,static_vector_size_types>::type;
61  using type = typename std::conditional<(std::numeric_limits<candidate_type>::max() >= Size),
62  candidate_type,
63  typename static_vector_size_type<Size,static_cast<std::size_t>(Index + 1u)>::type>::type;
64 };

100 class static_vector
101 {
102 #if !defined(PIRANHA_DOXYGEN_INVOKED)
103  static_assert(MaxSize > 0u,"Maximum size must be strictly positive.");

112  using size_type = typename detail::static_vector_size_type<MaxSize>::type;
113  private:
114 #if !defined(PIRANHA_DOXYGEN_INVOKED)
115  PIRANHA_TT_CHECK(is_container_element,T);
116  // This check is against overflows when using memcpy.
117  static_assert(std::numeric_limits<size_type>::max() <= std::numeric_limits<std::size_t>::max() / sizeof(T),
118  "The size type for static_vector might overflow.");
119  // Check for overflow in the definition of the storage type.
120  static_assert(MaxSize < std::numeric_limits<std::size_t>::max() / sizeof(T),"Overflow in the computation of storage size.");
121  // NOTE: some notes about the use of raw storage, after some research in the standard:
122  // - storage type is guaranteed to be a POD able to hold any object whose size is at most Len
123  // and alignment a divisor of Align (20.9.7.6). Thus, we can store on object of type T at the
124  // beginning of the storage;
125  // - POD types are trivially copyable, and thus they occupy contiguous bytes of storage (1.8/5);
126  // - the meaning of "contiguous" seems not to be explicitly stated, but it can be inferred
127  // (e.g., 23.3.2.1/1) that it has to do with the fact that one can do pointer arithmetics
128  // to move around in the storage;
129  // - the footnote in 5.7/6 implies that casting around to different pointer types and then doing
130  // pointer arithmetics has the expected behaviour of moving about with discrete offsets equal to the sizeof
131  // the pointed-to object (i.e., it seems like pointer arithmetics does not suffer from
132  // strict aliasing and type punning issues - at least as long as one goest through void *
133  // conversions to get the address of the storage - note also 3.9.2/3). This supposedly applies to arrays only,
134  // but, again, it is implied in other places that it also holds true for contiguous storage;
135  // - the no-padding guarantee for arrays (which also consist of contiguous storage) implies that sizeof must be
136  // a multiple of the alignment for a given type;
137  // - the definition of alignment (3.11) implies that if one has contiguous storage starting at an address in
138  // which T can be constructed, the offsetting the initial address by multiples of the alignment value will
139  // still produce addresses at which the object can be constructed;
140  // - in general, we are assuming here that we can handle contiguous storage the same way arrays can be handled (e.g.,
141  // to get the end() iterator we get one past the last element);
142  // - note that placement new will work as expected (i.e., it will construct the object exactly at the address passed
143  // in as parameter).
144  typedef typename std::aligned_storage<sizeof(T) * MaxSize,alignof(T)>::type storage_type;
145 #endif

169  static_vector(const static_vector &other):m_tag(1u),m_size(0u)
170  {
171  // NOTE: here and elsewhere, the standard implies (3.9/2) that we can use this optimisation
172  // for trivially copyable types. GCC does not support the type trait yet, so we restrict the
173  // optimisation to POD types (which are trivially copyable).
174  if (std::is_pod<T>::value) {
175  std::memcpy(vs(),other.vs(),other.m_size * sizeof(T));
176  m_size = other.m_size;
177  } else {
178  try {
179  const auto size = other.size();
180  for (size_type i = 0u; i < size; ++i) {
181  push_back(other[i]);
182  }
183  } catch (...) {
184  // Roll back the constructed items before re-throwing.
185  destroy_items();
186  throw;
187  }
188  }
189  }

194  static_vector(static_vector &&other) noexcept:m_tag(1u),m_size(0u)
195  {

198  std::memcpy(vs(),other.vs(),size * sizeof(T));
199  m_size = size;

201  // NOTE: here no need for rollback, as we assume move ctors do not throw.
202  for (size_type i = 0u; i < size; ++i) {

204  }
205  }

269  static_vector &operator=(static_vector &&other) noexcept
270  {
271  if (likely(this != &other)) {
272  if (std::is_pod<T>::value) {
273  std::memcpy(vs(),other.vs(),other.m_size * sizeof(T));
274  } else {
275  const size_type old_size = m_size, new_size = other.m_size;
276  if (old_size <= new_size) {
277  // Move assign new content into old content.
278  for (size_type i = 0u; i < old_size; ++i) {
279  (*this)[i] = std::move(other[i]);
280  }
281  // Move construct remaining elements.
282  for (size_type i = old_size; i < new_size; ++i) {
283  ::new (static_cast<void *>(ptr() + i)) value_type(std::move(other[i]));
284  }
285  } else {
286  // Move assign new content into old content.
287  for (size_type i = 0u; i < new_size; ++i) {
288  (*this)[i] = std::move(other[i]);
289  }
290  // Destroy excess content.
291  for (size_type i = new_size; i < old_size; ++i) {
292  ptr()[i].~T();
293  }
294  }
295  }
296  m_size = other.m_size;
297  // Nuke the other.
298  other.destroy_items();
299  other.m_size = 0u;
300  }
301  return *this;
302  }

464  void resize(const size_type &new_size)
465  {
466  if (unlikely(new_size > MaxSize)) {
467  piranha_throw(std::bad_alloc,);
468  }
469  const auto old_size = m_size;
470  if (new_size == old_size) {
471  return;
472  } else if (new_size > old_size) {
473  size_type i = old_size;
474  // Construct new in case of larger size.
475  try {
476  for (; i < new_size; ++i) {
477  // NOTE: placement-new syntax for value initialization, the same as performed by
478  // std::vector (see 23.3.6.2 and 23.3.6.3/9).
479  // http://en.cppreference.com/w/cpp/language/value_initialization
480  ::new (static_cast<void *>(ptr() + i)) value_type();
481  piranha_assert(m_size != MaxSize);
482  ++m_size;
483  }
484  } catch (...) {
485  for (size_type j = old_size; j < i; ++j) {
486  ptr()[j].~T();
487  piranha_assert(m_size);
488  --m_size;
489  }
490  throw;
491  }
492  } else {
493  // Destroy in case of smaller size.
494  for (size_type i = new_size; i < old_size; ++i) {
495  ptr()[i].~T();
496  piranha_assert(m_size);
497  --m_size;
498  }
499  }
500  }

513  iterator erase(const_iterator it)
514  {
515  // NOTE: on the use of const_cast: the object 'it' points to is *not* const by definition
516  // (it is an element of a vector). So the const casting should be ok. See these links:
517  // http://stackoverflow.com/questions/30770944/is-this-a-valid-usage-of-const-cast
518  // http://en.cppreference.com/w/cpp/language/const_cast
519  // All of this is noexcept.
520  piranha_assert(it != end());
521  // The return value is the original iterator, which will eventually contain the value
522  // following the original 'it'. If 'it' pointed to the last element, then retval
523  // will be the new end().
524  auto retval = const_cast<iterator>(it);
525  const auto it_f = end() - 1;
526  // NOTE: POD optimisation is possible here, but requires pointer diff.
527  for (; it != it_f; ++it) {
528  // Destroy the current element.
529  it->~T();
530  // Move-init the current iterator content with the next element.
531  ::new (static_cast<void *>(const_cast<iterator>(it))) value_type(std::move(*(it + 1)));
532  }
533  // Destroy the last element.
534  it->~T();
535  // Update the size.
536  m_size = static_cast<size_type>(m_size - 1u);
537  return retval;
538  }
