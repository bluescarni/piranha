433  bool is_compatible(const symbol_set &args) const noexcept
434  {
435  const auto s = args.size();
436  // No args means the value must also be zero.
437  if (s == 0u) {
438  return !m_value;
439  }
440  const auto &limits = ka::get_limits();
441  // If we overflow the maximum size available, we cannot use this object as key in series.
442  if (s >= limits.size()) {
443  return false;
444  }
445  const auto &l = limits[s];
446  // Value is compatible if it is within the bounds for the given size.
447  if (m_value < std::get<1u>(l) || m_value > std::get<2u>(l)) {
448  return false;
449  }
450  // Now check for the first multiplier.
451  // NOTE: here we have already checked all the conditions that could lead to unpack() throwing, so
452  // we do not need to put @throw specifications in the doc.
453  const auto unpacked = unpack(args);
454  // We know that s != 0.
455  piranha_assert(unpacked.size() > 0u);
456  for (typename v_type::size_type i = 0u; i < s; ++i) {
457  if (unpacked[i] < value_type(0)) {
458  return false;
459  } else if (unpacked[i] > value_type(0)) {
460  break;
461  }
462  }
463  return true;
464  }

666  template <typename Cf, multiply_enabler<Cf> = 0>
667  static void multiply(std::array<term<Cf,real_trigonometric_kronecker_monomial>,multiply_arity> &res,
668  const term<Cf,real_trigonometric_kronecker_monomial> &t1, const term<Cf,real_trigonometric_kronecker_monomial> &t2,
669  const symbol_set &args)
670  {
671  // Coefficients first.
672  // NOTE: first divide by two here and then multiply, at least for series? Would reduce
673  // quadratically the number of coefficient divisions.
674  // Similarly, it might be worth to change the sign as needed before the multiplication.
675  // NOTE: and if we divide everything by two in the end instead? Might be better overall.
676  Cf res_cf(t1.m_cf * t2.m_cf);
677  res_cf /= 2;
678  res[0u].m_cf = res_cf;
679  res[1u].m_cf = std::move(res_cf);
680  const bool f1 = t1.m_key.get_flavour(), f2 = t2.m_key.get_flavour();
681  if (f1 && f2) {
682  // cos, cos: no change.
683  } else if (!f1 && !f2) {
684  // sin, sin: negate the plus.
685  math::negate(res[0u].m_cf);
686  } else if (!f1 && f2) {
687  // sin, cos: no change.
688  } else {
689  // cos, sin: negate the minus.
690  math::negate(res[1u].m_cf);
691  }
692  // Now the keys.
693  auto &retval_plus = res[0u].m_key;
694  auto &retval_minus = res[1u].m_key;
695  // Flags to signal if a sign change in the multipliers was needed as part of the canonicalization.
696  bool sign_plus = false, sign_minus = false;
697  const auto size = args.size();
698  const auto tmp1 = t1.m_key.unpack(args), tmp2 = t2.m_key.unpack(args);
699  v_type result_plus, result_minus;
700  for (typename v_type::size_type i = 0u; i < size; ++i) {
701  result_plus.push_back(tmp1[i]);
702  detail::km_safe_adder(result_plus[i],tmp2[i]);
703  // NOTE: it is safe here to take the negative because in kronecker_array we are guaranteed
704  // that the range of each element is symmetric, so if tmp2[i] is representable also -tmp2[i] is.
705  // NOTE: the static cast here is because if value_type is narrower than int, the unary minus will promote
706  // to int and safe_adder won't work as it expects identical types.
707  result_minus.push_back(tmp1[i]);
708  detail::km_safe_adder(result_minus[i],static_cast<value_type>(-tmp2[i]));
709  }
710  // Handle sign changes.
711  sign_plus = canonicalise_impl(result_plus);
712  sign_minus = canonicalise_impl(result_minus);
713  // Compute them before assigning, so in case of exceptions we do not touch the return values.
714  const auto re_plus = ka::encode(result_plus), re_minus = ka::encode(result_minus);
715  retval_plus.m_value = re_plus;
716  retval_minus.m_value = re_minus;
717  const bool f = (t1.m_key.get_flavour() == t2.m_key.get_flavour());
718  retval_plus.m_flavour = f;
719  retval_minus.m_flavour = f;
720  // If multiplier sign was changed and the result is a sine, negate the coefficient.
721  if (sign_plus && !res[0u].m_key.get_flavour()) {
722  math::negate(res[0u].m_cf);
723  }
724  if (sign_minus && !res[1u].m_key.get_flavour()) {
725  math::negate(res[1u].m_cf);
726  }
727  }

782  void print(std::ostream &os, const symbol_set &args) const
783  {
784  // Don't print anything in case all multipliers are zero.
785  if (m_value == value_type(0)) {
786  return;
787  }
788  if (m_flavour) {
789  os << "cos(";
790  } else {
791  os << "sin(";
792  }
793  const auto tmp = unpack(args);
794  piranha_assert(tmp.size() == args.size());
795  const value_type zero(0), one(1), m_one(-1);
796  bool empty_output = true;
797  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
798  if (tmp[i] != zero) {
799  // A positive multiplier, in case previous output exists, must be preceded
800  // by a "+" sign.
801  if (tmp[i] > zero && !empty_output) {
802  os << '+';
803  }
804  // Print the multiplier, unless it's "-1": in that case, just print the minus sign.
805  if (tmp[i] == m_one) {
806  os << '-';
807  } else if (tmp[i] != one) {
808  os << detail::prepare_for_print(tmp[i]) << '*';
809  }
810  // Finally, print name of variable.
811  os << args[i].get_name();
812  empty_output = false;
813  }
814  }
815  os << ")";
816  }

882  auto v = unpack(args);
883  // Cannot take derivative wrt more than one variable, and the position of that variable
884  // must be compatible with the monomial.
885  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
886  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
887  }
888  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
889  // variable with zero multiplier.
890  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
891  // there is a valid position and hence the size must be not zero.
892  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
893  return std::make_pair(T(0),real_trigonometric_kronecker_monomial(args));
894  }
895  auto v_b = v.begin();
896  // Original multiplier.
897  T n(v_b[*p.begin()]);
898  // Create a copy of this.
899  real_trigonometric_kronecker_monomial tmp_m(*this);
900  // Flip the flavour.
901  tmp_m.set_flavour(!get_flavour());
902  // Flip the sign of the multiplier as needed.
903  if (get_flavour()) {
904  // NOTE: this is safe, as it is coming out of a k codification which
905  // is symmetric.
906  math::negate(n);
907  }
908  return std::make_pair(std::move(n),std::move(tmp_m));
909  }

930  std::pair<T,real_trigonometric_kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
931  {
932  auto v = unpack(args);
933  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i) {
934  if (args[i] == s && !math::is_zero(v[i])) {
935  integer tmp_n(v[i]);
936  real_trigonometric_kronecker_monomial tmp_m(*this);
937  // Flip the flavour.
938  tmp_m.set_flavour(!get_flavour());
939  // Flip the sign of the multiplier as needed.
940  if (!get_flavour()) {
941  tmp_n.negate();
942  }
943  return std::make_pair(static_cast<value_type>(tmp_n),std::move(tmp_m));
944  }
945 
946  }

973  template <typename U>
974  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
975  {
976  using return_type = eval_type<U>;
977  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
978  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
979  }
980  // Run the unpack before the checks below in order to check the suitability of args.
981  auto v = unpack(args);
982  if (args.size() == 0u) {
983  if (get_flavour()) {
984  return return_type(1);
985  } else {
986  return return_type(0);
987  }
988  }
989  using tmp_type = decltype(std::declval<U const &>() * std::declval<value_type const &>());
990  tmp_type tmp(0);
991  auto it = pmap.begin();
992  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i, ++it) {
993  piranha_assert(it != pmap.end() && it->first == i);
994  // NOTE: here it might make sense to use multiply_accumulate. There might be a perf gain
995  // for things like real. Maybe fast FMA on Haswell too. Remember to adapt the type requirements
996  // in case we implement this.
997  tmp += it->second * v[i];
998  }
999  piranha_assert(it == pmap.end());
1000  if (get_flavour()) {
1001  return math::cos(tmp);
1002  }
1003  return math::sin(tmp);
1004  }

1112  template <typename U>
1113  std::vector<std::pair<t_subs_type<U>,
1114  real_trigonometric_kronecker_monomial>> t_subs(const std::string &name, const U &c, const U &s, const symbol_set &args) const
1115  {
1116  typedef decltype(this->t_subs(name,c,s,args)) ret_type;
1117  typedef typename ret_type::value_type::first_type res_type;
1118  const auto v = unpack(args);
1119  v_type new_v;
1120  value_type n(0);
1121  // Build the new vector key.
1122  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i) {
1123  if (args[i].get_name() == name) {
1124  new_v.push_back(value_type(0));
1125  n = v[i];
1126  } else {
1127  new_v.push_back(v[i]);
1128  }
1129  }
1130  // Absolute value of the multiplier.
1131  const value_type abs_n = (n >= 0) ? n : static_cast<value_type>(-n);
1132  // Prepare the powers of c and s to be used in the multiple angles formulae.
1133  std::unordered_map<value_type,U> c_map, s_map;
1134  c_map[0] = U(1);
1135  s_map[0] = U(1);
1136  for (value_type k(0); k < abs_n; ++k) {
1137  c_map[k + value_type(1)] = c_map[k] * c;
1138  s_map[k + value_type(1)] = s_map[k] * s;
1139  }
1140  // Init with the first element in the summation.
1141  res_type cos_nx((cos_phase(abs_n) * math::binomial(abs_n,value_type(0))) * (c_map[value_type(0)] * s_map[abs_n])),
1142  sin_nx((sin_phase(abs_n) * math::binomial(abs_n,value_type(0))) * (c_map[value_type(0)] * s_map[abs_n]));
1143  for (value_type k(0); k < abs_n; ++k) {
1144  const value_type p = abs_n - (k + value_type(1));
1145  piranha_assert(p >= value_type(0));
1146  // NOTE: here the type is slightly different from the decltype() that determines the return type, but as long
1147  // as binomial(value_type,value_type) returns integer there will be no difference because of the
1148  // left-to-right associativity of multiplication.
1149  res_type tmp(math::binomial(abs_n,k + value_type(1)) * (c_map[k + value_type(1)] * s_map[p]));
1150  cos_nx += cos_phase(p) * tmp;
1151  sin_nx += sin_phase(p) * tmp;
1152  }
1153  // Change sign as necessary.
1154  if (abs_n != n) {
1155  math::negate(sin_nx);
1156  }
1157  // Buld the new keys and canonicalise as needed.
1158  const bool sign_changed = canonicalise_impl(new_v);
1159  piranha_assert(new_v.size() == v.size());
1160  const auto new_int = ka::encode(new_v);
1161  real_trigonometric_kronecker_monomial cos_key(new_int,true), sin_key(new_int,false);
1162  ret_type retval;
1163  if (get_flavour()) {
1164  retval.push_back(std::make_pair(std::move(cos_nx),std::move(cos_key)));
1165  retval.push_back(std::make_pair(std::move(sin_nx),std::move(sin_key)));
1166  // Need to flip the sign on the sin * sin product if sign was not changed.
1167  if (!sign_changed) {
1168  math::negate(retval[1u].first);
1169  }
1170  } else {
1171  retval.push_back(std::make_pair(std::move(sin_nx),std::move(cos_key)));
1172  retval.push_back(std::make_pair(std::move(cos_nx),std::move(sin_key)));
1173  // Need to flip the sign on the cos * sin product if sign was changed.
1174  if (sign_changed) {
1175  math::negate(retval[1u].first);
1176  }
1177  }
1178  return retval;
1179  }
