52 class cache_aligning_allocator: public dynamic_aligning_allocator<T>
53 {
54  using base = dynamic_aligning_allocator<T>;
55  static std::size_t determine_alignment()
56  {
57 #if !defined(PIRANHA_HAVE_MEMORY_ALIGNMENT_PRIMITIVES)
58  return 0u;
59 #endif
60  try {
61  const std::size_t alignment = boost::numeric_cast<std::size_t>(settings::get_cache_line_size());
62  if (!alignment_check<T>(alignment)) {
63  return 0u;
64  }
65  return alignment;
66  } catch (...) {
67  return 0u;
68  }
69  }
