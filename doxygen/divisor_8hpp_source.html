151  // Canonical term: the first nonzero element is positive and all the gcd of all elements is 1 or -1.
152  // NOTE: this also includes the check for all zero elements, as gcd(0,0,...,0) = 0.
153  static bool term_is_canonical(const p_type &p)
154  {
155  bool first_nonzero_found = false;
156  value_type cd(0);
157  for (const auto &n: p.v) {
158  if (!first_nonzero_found && !math::is_zero(n)) {
159  if (n < 0) {
160  return false;
161  }
162  first_nonzero_found = true;
163  }
164  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
165  cd = detail::gcd(cd,n);
166  }
167  if (cd != 1 && cd != -1) {
168  return false;
169  }
170  return true;
171  }

186  bool destruction_checks() const
187  {
188  const auto it_f = m_container.end();
189  auto it = m_container.begin();
190  const typename v_type::size_type v_size = (it == it_f) ? 0u : it->v.size();
191  for (; it != it_f; ++it) {
192  // Check: the exponent must be greater than zero.
193  if (it->e <= 0) {
194  return false;
195  }

198  return false;

201  if (!term_is_canonical(*it)) {

211  // Insertion machinery.
212  template <typename Term>
213  void insertion_impl(Term &&term)
214  {
215  // Handle the case of a table with no buckets.
216  if (unlikely(!m_container.bucket_count())) {
217  m_container._increase_size();
218  }
219  // Try to locate the term.
220  auto bucket_idx = m_container._bucket(term);
221  const auto it = m_container._find(term,bucket_idx);
222  if (it == m_container.end()) {
223  // New term.
224  if (unlikely(m_container.size() == std::numeric_limits<size_type>::max())) {
225  piranha_throw(std::overflow_error,"maximum number of elements reached");
226  }
227  // Term is new. Handle the case in which we need to rehash because of load factor.
228  if (unlikely(static_cast<double>(m_container.size() + size_type(1u)) / static_cast<double>(m_container.bucket_count()) >
229  m_container.max_load_factor()))
230  {
231  m_container._increase_size();
232  // We need a new bucket index in case of a rehash.
233  bucket_idx = m_container._bucket(term);
234  }
235  // Actually perform the insertion and finish by updating the size.
236  m_container._unique_insert(std::forward<Term>(term),bucket_idx);
237  m_container._update_size(m_container.size() + size_type(1u));
238  } else {
239  // Existing term - update the exponent.
240  update_exponent(it->e,term.e);
241  }
242  }
243  template <typename U = T, typename std::enable_if<std::is_integral<U>::value,int>::type = 0>
244  static void update_exponent(value_type &a, const value_type &b)
245  {
246  // Check the range when the exponent is an integral type.
247  piranha_assert(a > 0);
248  piranha_assert(b > 0);
249  // NOTE: this is safe as we require b to be a positive value.
250  if (unlikely(a > std::numeric_limits<value_type>::max() - b)) {
251  piranha_throw(std::invalid_argument,"overflow in the computation of the exponent of a divisor term");
252  }
253  a = static_cast<value_type>(a + b);
254  }
255  template <typename U = T, typename std::enable_if<!std::is_integral<U>::value,int>::type = 0>

257  {
258  // For integer exponents, just do normal addition.
259  a += b;

273  template <class Archive>
274  void load(Archive &ar, unsigned int)
275  {
276  divisor new_d;
277  try {
278  // NOTE: here we could throw either because the archive is garbage (Boost.Serialization throwing
279  // in this case) or because the loaded terms are invalid. In the second case we must make sure
280  // to destroy the content of new_d before exiting, otherwise the dtor of the divisor will
281  // hit assertion failures in debug mode. We need the second clear() in the catch block
282  // because we could be in the situation that the archive contains invaild terms and it is garbage
283  // after a while - in thie case invalid terms coud be living in new_d and the first clear() is never
284  // hit.
285  ar & new_d.m_container;
286  // Run the destruction checks, if they fail throw.
287  if (unlikely(!new_d.destruction_checks())) {
288  new_d.clear();
289  piranha_throw(std::invalid_argument,"error during the deserialization of a divisor, the loaded data "
290  "is invalid");
291  }
292  } catch (...) {
293  new_d.clear();
294  throw;
295  }
296  // Move in.
297  *this = std::move(new_d);

394  template <typename It, typename Exponent, insert_enabler<It,Exponent> = 0>
395  void insert(It begin, It end, const Exponent &e)
396  {
397  p_type term;
398  // Assign the exponent.
399  term.e = safe_cast<value_type>(e);
400  if (unlikely(term.e <= 0)) {
401  piranha_throw(std::invalid_argument,"a term of a divisor must have a positive exponent");
402  }
403  // Build the vector to be inserted.
404  for (; begin != end; ++begin) {
405  term.v.push_back(safe_cast<value_type>(*begin));
406  }
407  // Range check.
408  if (unlikely(!term_range_check(term))) {
409  piranha_throw(std::invalid_argument,"an element in a term of a divisor is outside the allowed range");
410  }
411  // Check that the term is canonical.
412  if (unlikely(!term_is_canonical(term))) {
413  piranha_throw(std::invalid_argument,"term not in canonical form");
414  }
415  // Perform the insertion.
416  insertion_impl(std::move(term));
417  }

476  std::size_t hash() const
477  {
478  // NOTE: here we are using a simple sum to compute the hash of a vector of vectors. Indeed,
479  // we need some add/multiply like operation in order to make sure that two divisors
480  // with the same elements stored in different order hash to the same thing. This seems
481  // to be ok, according to this,
482  // http://stackoverflow.com/questions/18021643/hashing-a-set-of-integers-in-an-order-independent-way,
483  // as the hash of each subvector should be rather good (it comes from boost hash combine).
484  // XOR is also commutative and might be used (see link above).
485  // If problem arises, we can consider muliplying the hash of each subvector by a prime
486  // (similar to Kronecker substitution).
487  std::size_t retval = 0u;
488  p_type_hasher hasher;
489  const auto it_f = m_container.end();
490  for (auto it = m_container.begin(); it != it_f; ++it) {
491  retval = static_cast<std::size_t>(retval + hasher(*it));
492  }
493  return retval;
494  }

588  void print(std::ostream &os, const symbol_set &args) const
589  {
590  // Don't print anything if there are no terms.
591  if (m_container.empty()) {
592  return;
593  }
594  if (unlikely(m_container.begin()->v.size() != args.size())) {
595  piranha_throw(std::invalid_argument,"invalid size of arguments set");
596  }
597  const auto it_f = m_container.end();
598  bool first_term = true;
599  os << "1/[";
600  for (auto it = m_container.begin(); it != it_f; ++it) {
601  // If this is not the first term, print a leading '*' operator.
602  if (first_term) {
603  first_term = false;
604  } else {
605  os << '*';
606  }
607  bool printed_something = false;
608  os << '(';
609  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i) {
610  // If the aij is zero, don't print anything.
611  if (math::is_zero(it->v[i])) {
612  continue;
613  }
614  // A positive aij, in case previous output exists, must be preceded
615  // by a "+" sign.
616  if (it->v[i] > 0 && printed_something) {
617  os << '+';
618  }
619  // Print the aij, unless it's "-1": in that case, just print the minus sign.
620  if (it->v[i] == -1) {
621  os << '-';
622  } else if (it->v[i] != 1) {
623  os << detail::prepare_for_print(it->v[i]) << '*';
624  }
625  // Finally, print name of variable.
626  os << args[i].get_name();
627  printed_something = true;
628  }
629  os << ')';
630  // Print the exponent, if different from one.
631  if (it->e != 1) {
632  os << "**" << detail::prepare_for_print(it->e);
633  }
634  }
635  os << ']';
636  }

710  template <typename U>
711  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
712  {
713  // Just return 1 if the container is empty.
714  eval_type<U> retval(1);
715  if (m_container.empty()) {
716  return retval;
717  }
718  // NOTE: the positions map must have the same number of elements as this, and it must
719  // be made of consecutive positions [0,size - 1].
720  if (unlikely(pmap.size() != m_container.begin()->v.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
721  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
722  }
723  // Args must be compatible with this.
724  if (unlikely(args.size() != m_container.begin()->v.size())) {
725  piranha_throw(std::invalid_argument,"invalid size of arguments set");
726  }
727  const auto it_f = m_container.end();
728  for (auto it = m_container.begin(); it != it_f; ++it) {
729  eval_sum_type<U> tmp(0);
730  auto map_it = pmap.begin();
731  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i, ++map_it) {
732  piranha_assert(map_it != pmap.end() && map_it->first == i);
733  // NOTE: might use multadd here eventually for performance.
734  tmp += it->v[i] * map_it->second;
735  }
736  piranha_assert(map_it == pmap.end());
737  // NOTE: consider rewriting this in terms of multiplications as a performance
738  // improvement - the eval_type deduction should be changed accordingly.
739  retval /= math::pow(tmp,it->e);
740  }
741  return retval;
742  }

769  const term<Cf,divisor> &t2, const symbol_set &args)
