391  bool is_compatible(const symbol_set &args) const noexcept
392  {
393  // NOTE: the idea here is to avoid unpack()ing for performance reasons: these checks
394  // are already part of unpack(), and that's why unpack() is used instead of is_compatible()
395  // in other methods.
396  const auto s = args.size();
397  // No args means the value must also be zero.
398  if (s == 0u) {
399  return !m_value;
400  }
401  const auto &limits = ka::get_limits();
402  // If we overflow the maximum size available, we cannot use this object as key in series.
403  if (s >= limits.size()) {
404  return false;
405  }
406  const auto &l = limits[static_cast<decltype(limits.size())>(s)];
407  // Value is compatible if it is within the bounds for the given size.
408  return (m_value >= std::get<1u>(l) && m_value <= std::get<2u>(l));
409  }

592  {

697  void print_tex(std::ostream &os, const symbol_set &args) const
698  {
699  const auto tmp = unpack(args);
700  std::ostringstream oss_num, oss_den, *cur_oss;
701  const value_type zero(0), one(1);
702  value_type cur_value;
703  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
704  cur_value = tmp[i];

706  // NOTE: here negate() is safe because of the symmetry in kronecker_array.
707  cur_oss = (cur_value > zero) ? std::addressof(oss_num) : (math::negate(cur_value),std::addressof(oss_den));
708  (*cur_oss) << "{" << args[i].get_name() << "}";
709  if (cur_value != one) {
710  (*cur_oss) << "^{" << static_cast<long long>(cur_value) << "}";
711  }
712  }
713  }
714  const std::string num_str = oss_num.str(), den_str = oss_den.str();
715  if (!num_str.empty() && !den_str.empty()) {
716  os << "\\frac{" << num_str << "}{" << den_str << "}";
717  } else if (!num_str.empty() && den_str.empty()) {
718  os << num_str;
719  } else if (num_str.empty() && !den_str.empty()) {
720  os << "\\frac{1}{" << den_str << "}";

742  std::pair<T,kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
743  {
744  auto v = unpack(args);
745  // Cannot take derivative wrt more than one variable, and the position of that variable
746  // must be compatible with the monomial.
747  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
748  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
749  }
750  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
751  // variable with zero exponent.
752  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
753  // there is a valid position and hence the size must be not zero.
754  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
755  return std::make_pair(T(0),kronecker_monomial(args));
756  }
757  auto v_b = v.begin();
758  // Original exponent.
759  T n(v_b[*p.begin()]);
760  // Decrement the exponent in the monomial.
761  if (unlikely(n == std::numeric_limits<T>::min())) {
762  piranha_throw(std::invalid_argument,"negative overflow error in the calculation of the "
763  "partial derivative of a monomial");
764  }
765  v_b[*p.begin()] = static_cast<T>(n - T(1));
766  kronecker_monomial tmp_km;
767  tmp_km.m_value = ka::encode(v);
768  return std::make_pair(n,std::move(tmp_km));
769  }

791  std::pair<T,kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
792  {
793  v_type v = unpack(args), retval;
794  value_type expo(0), one(1);
795  for (min_int<typename v_type::size_type,decltype(args.size())> i = 0u; i < args.size(); ++i) {
796  if (math::is_zero(expo) && s < args[i]) {
797  // If we went past the position of s in args and still we
798  // have not performed the integration, it means that we need to add
799  // a new exponent.
800  retval.push_back(one);
801  expo = one;
802  }
803  retval.push_back(v[i]);
804  if (args[i] == s) {
805  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
806  // we are never going to land here as args[i] is at this point never going to be s.
807  if (unlikely(retval[i] == std::numeric_limits<value_type>::max())) {
808  piranha_throw(std::invalid_argument,"positive overflow error in the calculation of the integral of a monomial");
809  }
810  retval[i] = static_cast<value_type>(retval[i] + value_type(1));
811  if (math::is_zero(retval[i])) {
812  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
813  }
814  expo = retval[i];
815  }
816  }
817  // If expo is still zero, it means we need to add a new exponent at the end.
818  if (math::is_zero(expo)) {
819  retval.push_back(one);
820  expo = one;
821  }
822  return std::make_pair(expo,kronecker_monomial(ka::encode(retval)));
823  }

849  template <typename U>
850  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
851  {
852  using return_type = eval_type<U>;
853  using size_type = typename v_type::size_type;
854  // NOTE: here we can check the pmap size only against args.
855  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
856  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
857  }
858  auto v = unpack(args);
859  return_type retval(1);
860  auto it = pmap.begin();
861  for (min_int<size_type,decltype(args.size())> i = 0u; i < args.size(); ++i, ++it) {
862  piranha_assert(it != pmap.end() && it->first == i);
863  retval *= math::pow(it->second,v[i]);
864  }
865  piranha_assert(it == pmap.end());
866  return retval;
