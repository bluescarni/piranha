178  template <typename Functor>
179  Series execute() const
180  {
181  // Do not do anything if one of the two series is empty.
182  if (unlikely(m_s1->empty() || m_s2->empty())) {
183  // NOTE: requirement is ok, a series must be def-ctible.
184  return Series{};
185  }
186  // This is the size type that will be used throughout the calculations.
187  using size_type = decltype(m_v1.size());
188  const size_type size1 = m_v1.size(), size2 = m_v2.size();
189  piranha_assert(size1 && size2);
190  // Establish the number of threads to use.
191  size_type n_threads = safe_cast<size_type>(thread_pool::use_threads(
192  integer(size1) * size2,integer(settings::get_min_work_per_thread())
193  ));
194  piranha_assert(n_threads);
195  // An additional check on n_threads is that its size is not greater than the size of the first series,
196  // as we are using the first operand to split up the work.
197  if (n_threads > size1) {
198  n_threads = size1;
199  }
200  piranha_assert(n_threads >= 1u);
201  if (likely(n_threads == 1u)) {
202  Series retval;
203  retval.m_symbol_set = m_s1->m_symbol_set;
204  Functor f(&m_v1[0u],size1,&m_v2[0u],size2,retval);
205  const auto tmp = rehasher(f);
206  blocked_multiplication(f);
207  if (tmp.first) {
208  trace_estimates(retval.size(),tmp.second);
209  }
210  return retval;
211  } else {
212  // Build the return values and the multiplication functors.
213  std::list<Series,cache_aligning_allocator<Series>> retval_list;
214  std::list<Functor,cache_aligning_allocator<Functor>> functor_list;
215  const auto block_size = size1 / n_threads;
216  for (size_type i = 0u; i < n_threads; ++i) {
217  // Last thread needs a different size from block_size.
218  const size_type s1 = (i == n_threads - 1u) ? (size1 - i * block_size) : block_size;
219  // NOTE: ok, series must be def-ctible.
220  retval_list.emplace_back();
221  retval_list.back().m_symbol_set = m_s1->m_symbol_set;
222  functor_list.push_back(Functor(&m_v1[0u] + i * block_size,s1,&m_v2[0u],size2,retval_list.back()));
223  }
224  auto f_it = functor_list.begin();
225  auto r_it = retval_list.begin();
226  future_list<std::future<void>> f_list;
227  try {
228  for (size_type i = 0u; i < n_threads; ++i, ++f_it, ++r_it) {
229  // Functor for use in the thread.
230  auto f = [f_it,r_it,this]() {
231  const auto tmp = this->rehasher(*f_it);
232  this->blocked_multiplication(*f_it);
233  if (tmp.first) {
234  this->trace_estimates(r_it->m_container.size(),tmp.second);
235  }
236  };
237  f_list.push_back(thread_pool::enqueue(static_cast<unsigned>(i),f));
238  }
239  f_list.wait_all();
240  f_list.get_all();
241  } catch (...) {
242  f_list.wait_all();
243  throw;
244  }
245  Series retval;
246  retval.m_symbol_set = m_s1->m_symbol_set;
247  auto final_estimate = estimate_final_series_size(Functor(&m_v1[0u],size1,&m_v2[0u],size2,retval));
248  // We want to make sure that final_estimate contains at least 1 element, so that we can use faster low-level
249  // methods in hash_set.
250  if (unlikely(!final_estimate)) {
251  final_estimate = 1u;
252  }
253  // Try to see if a series already has enough buckets.
254  auto it = std::find_if(retval_list.begin(),retval_list.end(),[&final_estimate](const Series &r) {
255  return static_cast<double>(r.m_container.bucket_count()) * r.m_container.max_load_factor() >= final_estimate;
256  });
257  if (it != retval_list.end()) {
258  retval = std::move(*it);
259  retval_list.erase(it);
260  } else {
261  // Otherwise, just rehash to the desired value, corrected for max load factor.
262  retval.m_container.rehash(
263  safe_cast<typename Series::size_type>(std::ceil(static_cast<double>(final_estimate) / retval.m_container.max_load_factor()))
264  );
265  }
266  // Cleanup functor that will erase all elements in retval_list.
267  auto cleanup = [&retval_list]() {
268  std::for_each(retval_list.begin(),retval_list.end(),[](Series &r) {r.m_container.clear();});
269  };
270  try {
271  final_merge(retval,retval_list,n_threads);
272  // Here the estimate will be correct if the initial estimate (corrected for load factor)
273  // is at least equal to the final real bucket count.
274  trace_estimates(retval.m_container.size(),final_estimate);
275  } catch (...) {
276  // Do the cleanup before re-throwing, as we use mutable iterators in final_merge.
277  cleanup();
278  // Clear also retval: it could have bogus number of elements if errors arise in the final merge.
279  retval.m_container.clear();
280  throw;
281  }
282  // Clean up the temporary series.
283  cleanup();
284  return retval;
285  }

378  template <typename Functor>
379  static void blocked_multiplication(const Functor &f)
380  {
381  typedef typename std::decay<decltype(f.m_s1)>::type size_type;
382  // NOTE: hard-coded block size of 256.
383  static_assert(std::numeric_limits<size_type>::max() >= 256u, "Invalid size type.");
384  const size_type size1 = f.m_s1, size2 = f.m_s2, bsize1 = 256u, nblocks1 = size1 / bsize1, bsize2 = bsize1, nblocks2 = size2 / bsize2;
385  // Start and end of last (possibly irregular) blocks.
386  const size_type i_ir_start = nblocks1 * bsize1, i_ir_end = size1;
387  const size_type j_ir_start = nblocks2 * bsize2, j_ir_end = size2;
388  for (size_type n1 = 0u; n1 < nblocks1; ++n1) {
389  const size_type i_start = n1 * bsize1, i_end = i_start + bsize1;
390  // regulars1 * regulars2
391  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
392  const size_type j_start = n2 * bsize2, j_end = j_start + bsize2;
393  for (size_type i = i_start; i < i_end; ++i) {
394  for (size_type j = j_start; j < j_end; ++j) {
395  f(i,j);
396  f.insert();
397  }
398  }
399  }
400  // regulars1 * rem2
401  for (size_type i = i_start; i < i_end; ++i) {
402  for (size_type j = j_ir_start; j < j_ir_end; ++j) {
403  f(i,j);
404  f.insert();
405  }
406  }
407  }
408  // rem1 * regulars2
409  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
410  const size_type j_start = n2 * bsize2, j_end = j_start + bsize2;
411  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
412  for (size_type j = j_start; j < j_end; ++j) {
413  f(i,j);
414  f.insert();
415  }
416  }
417  }
418  // rem1 * rem2.
419  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
420  for (size_type j = j_ir_start; j < j_ir_end; ++j) {
421  f(i,j);
422  f.insert();
423  }
424  }

443  template <typename Functor>
444  static typename Series::size_type estimate_final_series_size(const Functor &f)
445  {
446  typedef typename Series::size_type bucket_size_type;
447  typedef typename std::decay<decltype(f.m_s1)>::type size_type;
448  const size_type size1 = f.m_s1, size2 = f.m_s2;
449  // If one of the two series is empty, just return 0.
450  if (unlikely(!size1 || !size2)) {
451  return 0u;
452  }
453  // Cache reference to return series.
454  auto &retval = f.m_retval;
455  // NOTE: Hard-coded number of trials = 10.
456  // NOTE: here consider that in case of extremely sparse series with few terms (e.g., next to the lower limit
457  // for which this function is called) this will incur in noticeable overhead, since we will need many term-by-term
458  // before encountering the first duplicate.
459  const auto ntrials = 10u;
460  // NOTE: Hard-coded value for the estimation multiplier.
461  // NOTE: This value should be tuned for performance/memory usage tradeoffs.
462  const auto multiplier = 2;
463  // Size of the multiplication result
464  // Vectors of indices.
465  std::vector<size_type> v_idx1, v_idx2;
466  for (size_type i = 0u; i < size1; ++i) {
467  v_idx1.push_back(i);
468  }
469  for (size_type i = 0u; i < size2; ++i) {
470  v_idx2.push_back(i);
471  }
472  // Maxium number of random multiplications before which a duplicate term must be generated.
473  const size_type max_M = static_cast<size_type>(((integer(size1) * size2) / multiplier).sqrt());
474  // Random number engine.
475  std::mt19937 engine;
476  // Init counter.
477  integer total(0);
478  // Go with the trials.
479  // NOTE: This could be easily parallelised, but not sure if it is worth.
480  for (auto n = 0u; n < ntrials; ++n) {
481  // Randomise.
482  std::shuffle(v_idx1.begin(),v_idx1.end(),engine);
483  std::shuffle(v_idx2.begin(),v_idx2.end(),engine);
484  size_type count = 0u;
485  auto it1 = v_idx1.begin(), it2 = v_idx2.begin();
486  while (count < max_M) {
487  if (it1 == v_idx1.end()) {
488  // Each time we wrap around the first series,
489  // wrap around also the second one and rotate it.
490  it1 = v_idx1.begin();
491  auto middle = v_idx2.end();
492  --middle;
493  std::rotate(v_idx2.begin(),middle,v_idx2.end());
494  it2 = v_idx2.begin();
495  }
496  if (it2 == v_idx2.end()) {
497  it2 = v_idx2.begin();
498  }
499  // Perform multiplication and check if it produces a new term.
500  f(*it1,*it2);
501  f.insert();
502  constexpr std::size_t result_size = Series::term_type::key_type::multiply_arity;
503  // Check for unlikely overflow when increasing count.
504  if (unlikely(result_size > std::numeric_limits<size_type>::max() ||
505  count > std::numeric_limits<size_type>::max() - result_size))
506  {
507  piranha_throw(std::overflow_error,"overflow error");
508  }
509  if (retval.size() != count + result_size) {
510  break;
511  }
512  // Increase cycle variables.
513  count += result_size;
514  ++it1;
515  ++it2;
516  }
517  total += count;
518  // Reset retval.
519  retval.m_container.clear();
520  }
521  const auto mean = total / ntrials;
522  // Avoid division by zero.
523  if (total.sign()) {
524  const integer M = (mean * mean * multiplier * total) / total;
525  return static_cast<bucket_size_type>(M);
526  } else {
527  return static_cast<bucket_size_type>(mean * mean * multiplier);
528  }
529  }

549  static void trace_estimates(const typename Series::size_type &real_size, const typename Series::size_type &estimate)
550  {
551  if (unlikely(!real_size)) {
552  return;
553  }
554  tracing::trace("number_of_estimates",[](boost::any &x) {
555  if (unlikely(x.empty())) {
556  x = 0ull;
557  }
558  auto ptr = boost::any_cast<unsigned long long>(&x);
559  if (likely((bool)ptr)) {
560  ++*ptr;
561  }
562  });
563  tracing::trace("number_of_correct_estimates",[real_size,estimate](boost::any &x) {
564  if (unlikely(x.empty())) {
565  x = 0ull;
566  }
567  auto ptr = boost::any_cast<unsigned long long>(&x);
568  if (likely((bool)ptr)) {
569  *ptr += static_cast<unsigned long long>(estimate >= real_size);
570  }
571  });
572  tracing::trace("accumulated_estimate_ratio",[real_size,estimate](boost::any &x) {

574  x = 0.;
575  }
576  auto ptr = boost::any_cast<double>(&x);

578  *ptr += static_cast<double>(estimate) / static_cast<double>(real_size);
579  }

583  template <typename RetvalList, typename Size>
584  static void final_merge(Series &retval, RetvalList &retval_list, const Size &n_threads)
585  {
586  piranha_assert(n_threads > 1u);
587  piranha_assert(retval.m_container.bucket_count());
588  typedef typename std::vector<std::pair<typename Series::size_type,decltype(retval.m_container._m_begin())>> container_type;
589  typedef typename container_type::size_type size_type;
590  // First, let's fill a vector assigning each term of each element in retval_list to a bucket in retval.
591  const size_type size = static_cast<size_type>(
592  std::accumulate(retval_list.begin(),retval_list.end(),integer(0),[](const integer &n, const Series &r) {return n + r.size();}));
593  container_type idx(size);
594  future_list<std::future<void>> f_list1;
595  size_type i = 0u;
596  piranha_assert(retval_list.size() <= n_threads);
597  try {
598  unsigned thread_idx = 0u;
599  for (auto r_it = retval_list.begin(); r_it != retval_list.end(); ++r_it, ++thread_idx) {
600  auto f = [&idx,&retval,i,r_it]() {
601  const auto it_f = r_it->m_container._m_end();
602  // NOTE: size_type can represent the sum of the sizes of all retvals,
603  // so it will not overflow here.
604  size_type tmp_i = i;
605  for (auto it = r_it->m_container._m_begin(); it != it_f; ++it, ++tmp_i) {
606  piranha_assert(tmp_i < idx.size());
607  idx[tmp_i] = std::make_pair(retval.m_container._bucket(*it),it);
608  }
609  };
610  f_list1.push_back(thread_pool::enqueue(thread_idx,f));
611  i += r_it->size();
612  }
613  f_list1.wait_all();
614  f_list1.get_all();
615  } catch (...) {
616  f_list1.wait_all();
617  throw;
618  }
619  future_list<std::future<void>> f_list2;
620  std::mutex m;
621  std::vector<integer> new_sizes;
622  new_sizes.reserve(static_cast<std::vector<integer>::size_type>(n_threads));
623  if (unlikely(new_sizes.capacity() != n_threads)) {
624  piranha_throw(std::bad_alloc,);
625  }
626  const typename Series::size_type bucket_count = retval.m_container.bucket_count(), block_size = bucket_count / n_threads;
627  try {
628  for (Size n = 0u; n < n_threads; ++n) {
629  // These are the bucket indices allowed to the current thread.
630  const typename Series::size_type start = n * block_size, end = (n == n_threads - 1u) ? bucket_count : (n + 1u) * block_size;
631  auto f = [start,end,&size,&retval,&idx,&m,&new_sizes]() {
632  size_type count_plus = 0u, count_minus = 0u;
633  for (size_type i = 0u; i < size; ++i) {
634  const auto &bucket_idx = idx[i].first;
635  auto &term_it = idx[i].second;
636  if (bucket_idx >= start && bucket_idx < end) {
637  // Reconstruct part of series insertion, without the update in the number of elements.
638  // NOTE: compatibility and ignorability do not matter, as terms being inserted come from series
639  // anyway. Same for check on bucket_count.
640  const auto it = retval.m_container._find(*term_it,bucket_idx);
641  if (it == retval.m_container.end()) {
642  // Term is new.
643  retval.m_container._unique_insert(std::move(*term_it),bucket_idx);
644  // Update term count.
645  piranha_assert(count_plus < std::numeric_limits<size_type>::max());
646  ++count_plus;
647  } else {
648  // Assert the existing term is not ignorable and it is compatible.
649  piranha_assert(!it->is_ignorable(retval.m_symbol_set) && it->is_compatible(retval.m_symbol_set));
650  // Cleanup function.
651  auto cleanup = [&]() {
652  if (unlikely(it->is_ignorable(retval.m_symbol_set))) {
653  retval.m_container._erase(it);
654  // After term is erased, update count.
655  piranha_assert(count_minus < std::numeric_limits<size_type>::max());
656  ++count_minus;
657  }
658  };
659  try {
660  // The term exists already, update it.
661  retval.template insertion_cf_arithmetics<true>(it,std::move(*term_it));
662  // Check if the term has become ignorable after the modification.
663  cleanup();
664  } catch (...) {
665  cleanup();
666  throw;
667  }
668  }
669  }
670  }
671  // Store the new size.
672  std::lock_guard<std::mutex> lock(m);
673  new_sizes.push_back(integer(count_plus) - integer(count_minus));
674  };
675  // NOTE: here the static cast is safe as nthreads is coming from a call to
676  // the size of the thread pool.
677  f_list2.push_back(thread_pool::enqueue(static_cast<unsigned>(n),f));
678  }
679  f_list2.wait_all();
680  f_list2.get_all();
681  } catch (...) {
682  f_list2.wait_all();
683  throw;
684  }
685  // Final update of retval's size.
686  const auto new_size = std::accumulate(new_sizes.begin(),new_sizes.end(),integer(0));
687  piranha_assert(new_size + retval.m_container.size() >= 0);
