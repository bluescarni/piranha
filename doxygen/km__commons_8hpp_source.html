47 template <typename VType, typename KaType, typename T>
48 inline VType km_unpack(const symbol_set &args, const T &value)
49 {
50  if (unlikely(args.size() > VType::max_size)) {
51  piranha_throw(std::invalid_argument,"input set of arguments is too large for unpacking");
52  }
53  VType retval(static_cast<typename VType::size_type>(args.size()),0);
54  piranha_assert(args.size() == retval.size());
55  KaType::decode(retval,value);
56  return retval;
57 }

59 template <typename VType, typename KaType, typename T>
60 inline T km_merge_args(const symbol_set &orig_args, const symbol_set &new_args, const T &value)
61 {
62  using size_type = min_int<typename VType::size_type,decltype(new_args.size())>;
63  if (unlikely(new_args.size() <= orig_args.size() ||
64  !std::includes(new_args.begin(),new_args.end(),orig_args.begin(),orig_args.end())))
65  {
66  piranha_throw(std::invalid_argument,"invalid argument(s) for symbol set merging");
67  }
68  piranha_assert(std::is_sorted(orig_args.begin(),orig_args.end()));
69  piranha_assert(std::is_sorted(new_args.begin(),new_args.end()));
70  const auto old_vector = km_unpack<VType,KaType>(orig_args,value);
71  VType new_vector;
72  auto it_new = new_args.begin();
73  for (size_type i = 0u; i < old_vector.size(); ++i, ++it_new) {
74  while (*it_new != orig_args[i]) {
75  // NOTE: for arbitrary int types, value_type(0) might throw. Update docs
76  // if needed.
77  new_vector.push_back(T(0));
78  piranha_assert(it_new != new_args.end());
79  ++it_new;
80  piranha_assert(it_new != new_args.end());
81  }
82  new_vector.push_back(old_vector[i]);
83  }
84  // Fill up arguments at the tail of new_args but not in orig_args.
85  for (; it_new != new_args.end(); ++it_new) {
86  new_vector.push_back(T(0));
87  }
88  piranha_assert(new_vector.size() == new_args.size());
89  // Return new encoded value.
90  return KaType::encode(new_vector);
91 }

93 template <typename VType, typename KaType, typename T>
94 inline void km_trim_identify(symbol_set &candidates, const symbol_set &args, const T &value)
95 {
96  const VType tmp = km_unpack<VType,KaType>(args,value);
97  for (min_int<decltype(tmp.size()),decltype(args.size())> i = 0u; i < tmp.size(); ++i) {
98  if (!math::is_zero(tmp[i]) && std::binary_search(candidates.begin(),candidates.end(),args[i])) {
99  candidates.remove(args[i]);
100  }
101  }
102 }

104 template <typename VType, typename KaType, typename T>
105 inline T km_trim(const symbol_set &trim_args, const symbol_set &orig_args, const T &value)
106 {
107  using size_type = min_int<typename VType::size_type,decltype(orig_args.size())>;
108  const VType tmp = km_unpack<VType,KaType>(orig_args,value);
109  VType new_vector;
110  for (size_type i = 0u; i < tmp.size(); ++i) {
111  if (!std::binary_search(trim_args.begin(),trim_args.end(),orig_args[i])) {
112  new_vector.push_back(tmp[i]);
113  }
114  }
115  return KaType::encode(new_vector);
116 }
