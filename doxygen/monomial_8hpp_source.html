407  template <typename U = T, linarg_enabler<U> = 0>
408  std::string linear_argument(const symbol_set &args) const
409  {
410  if (!is_compatible(args)) {
411  piranha_throw(std::invalid_argument,"invalid size of arguments set");
412  }
413  typedef typename base::size_type size_type;
414  const size_type size = this->size();
415  size_type n_linear = 0u, candidate = 0u;
416  for (size_type i = 0u; i < size; ++i) {
417  integer tmp;
418  try {
419  tmp = safe_cast<integer>((*this)[i]);
420  } catch (const std::invalid_argument &) {
421  piranha_throw(std::invalid_argument,"exponent is not an integer");
422  }
423  if (tmp == 0) {
424  continue;
425  }
426  if (tmp != 1) {
427  piranha_throw(std::invalid_argument,"exponent is not unitary");
428  }
429  candidate = i;
430  ++n_linear;
431  }
432  if (n_linear != 1u) {
433  piranha_throw(std::invalid_argument,"monomial is not linear");
434  }
435  return args[static_cast<decltype(args.size())>(candidate)].get_name();

500  template <typename U = T, partial_enabler<U> = 0>
501  std::pair<U,monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
502  {
503  if (!is_compatible(args)) {
504  piranha_throw(std::invalid_argument,"invalid size of arguments set");
505  }
506  // Cannot take derivative wrt more than one variable, and the position of that variable
507  // must be compatible with the monomial.
508  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
509  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
510  }
511  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
512  // variable with zero exponent.
513  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
514  // there is a valid position and hence the size must be not zero.
515  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
516  return std::make_pair(U(0),monomial(args));
517  }
518  // Copy of the original monomial.
519  monomial m(*this);
520  auto m_b = m.begin();
521  // Original exponent.
522  U v(m_b[*p.begin()]);
523  // Decrement the exponent in the monomial.
524  ip_dec(m_b[*p.begin()]);
525  return std::make_pair(std::move(v),std::move(m));
526  }

557  template <typename U = T, integrate_enabler<U> = 0>
558  std::pair<U,monomial> integrate(const symbol &s, const symbol_set &args) const
559  {
560  typedef typename base::size_type size_type;
561  if (!is_compatible(args)) {
562  piranha_throw(std::invalid_argument,"invalid size of arguments set");
563  }
564  monomial retval;
565  T expo(0), one(1);
566  for (size_type i = 0u; i < args.size(); ++i) {
567  if (math::is_zero(expo) && s < args[i]) {
568  // If we went past the position of s in args and still we
569  // have not performed the integration, it means that we need to add
570  // a new exponent.
571  retval.push_back(one);
572  expo = one;
573  }
574  retval.push_back((*this)[i]);
575  if (args[i] == s) {
576  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
577  // we are never going to land here as args[i] is at this point never going to be s.
578  ip_inc(retval[i]);
579  if (math::is_zero(retval[i])) {
580  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
581  }
582  expo = retval[i];
583  }
584  }
585  // If expo is still zero, it means we need to add a new exponent at the end.
586  if (math::is_zero(expo)) {
587  retval.push_back(one);
588  expo = one;
589  }
590  return std::make_pair(std::move(expo),std::move(retval));
591  }

640  void print_tex(std::ostream &os, const symbol_set &args) const
641  {
642  if (unlikely(args.size() != this->size())) {
643  piranha_throw(std::invalid_argument,"invalid size of arguments set");
644  }
645  std::ostringstream oss_num, oss_den, *cur_oss;

649  cur_value = (*this)[i];
650  if (cur_value != zero) {
651  cur_oss = (cur_value > zero) ? std::addressof(oss_num) : (math::negate(cur_value),std::addressof(oss_den));

653  if (cur_value != one) {

656  }

658  const std::string num_str = oss_num.str(), den_str = oss_den.str();
659  if (!num_str.empty() && !den_str.empty()) {
660  os << "\\frac{" << num_str << "}{" << den_str << "}";
661  } else if (!num_str.empty() && den_str.empty()) {
662  os << num_str;
663  } else if (num_str.empty() && !den_str.empty()) {
664  os << "\\frac{1}{" << den_str << "}";

692  template <typename U>
693  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
694  {
695  using return_type = eval_type<U>;
696  using size_type = typename base::size_type;
697  // NOTE: the positions map must have the same number of elements as this, and it must
698  // be made of consecutive positions [0,size - 1].
699  if (unlikely(pmap.size() != this->size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
700  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
701  }
702  // Args must be compatible with this.
703  if (unlikely(args.size() != this->size())) {
704  piranha_throw(std::invalid_argument,"invalid size of arguments set");
705  }
706  return_type retval(1);
707  auto it = pmap.begin();
708  for (size_type i = 0u; i < this->size(); ++i, ++it) {
709  piranha_assert(it != pmap.end() && it->first == i);
710  retval *= math::pow(it->second,(*this)[i]);
711  }
712  piranha_assert(it == pmap.end());
713  return retval;

845  template <typename Cf, typename U = monomial, multiply_enabler<Cf,U> = 0>
