151  template <typename Float>
152  void construct_from_interoperable(const Float &x, typename std::enable_if<std::is_floating_point<Float>::value>::type * = nullptr)
153  {
154  if (unlikely(!std::isfinite(x))) {
155  piranha_throw(std::invalid_argument,"cannot construct a rational from a non-finite floating-point number");
156  }
157  // Denominator is always inited as 1.
158  m_den = 1;
159  if (x == Float(0)) {
160  // m_den is 1 already.
161  return;
162  }
163  Float abs_x = std::abs(x);
164  const unsigned radix = static_cast<unsigned>(std::numeric_limits<Float>::radix);
165  int_type i_part;
166  int exp = std::ilogb(abs_x);
167  while (exp >= 0) {
168  i_part += int_type(radix).pow(exp);
169  const Float tmp = std::scalbn(Float(1),exp);
170  if (unlikely(tmp == HUGE_VAL)) {
171  piranha_throw(std::invalid_argument,"output of std::scalbn is HUGE_VAL");
172  }
173  abs_x -= tmp;
174  // Break out if x is an exact integer.
175  if (abs_x == Float(0)) {
176  // m_den is 1 already.
177  m_num = i_part;
178  if (x < Float(0)) {
179  m_num.negate();
180  }
181  return;
182  }
183  exp = std::ilogb(abs_x);
184  if (unlikely(exp == INT_MAX || exp == FP_ILOGBNAN)) {
185  piranha_throw(std::invalid_argument,"error calling std::ilogb");
186  }
187  }
188  piranha_assert(abs_x < Float(1));
189  // Lift up the decimal part into an integer.
190  while (abs_x != Float(0)) {
191  abs_x = std::scalbln(abs_x,1);
192  if (unlikely(abs_x == HUGE_VAL)) {
193  piranha_throw(std::invalid_argument,"output of std::scalbn is HUGE_VAL");
194  }
195  const auto t_abs_x = std::trunc(abs_x);
196  m_den *= radix;
197  m_num *= radix;
198  // NOTE: here t_abs_x is guaranteed to be in
199  // [0,radix - 1], so the cast to unsigned should be ok.
200  // Note that floating-point numbers are guaranteed to be able
201  // to represent exactly at least a [-exp,+exp] exponent range
202  // (see the minimum values for the FLT constants in the C standard).
203  m_num += static_cast<unsigned>(t_abs_x);
204  abs_x -= t_abs_x;
205  }
206  math::multiply_accumulate(m_num,i_part,m_den);
207  canonicalise();
208  if (x < Float(0)) {
209  m_num.negate();
210  }
211  }

235  // In-place add.
236  mp_rational &in_place_add(const mp_rational &other)
237  {
238  // NOTE: all this should never throw because we only operate on mp_integer objects,
239  // no conversions involved, etc.
240  const bool u1 = m_den.is_unitary(), u2 = other.m_den.is_unitary();
241  if (u1 && u2) {
242  // Both are integers, just add without canonicalising. This is safe if
243  // this and other are the same object.
244  m_num += other.m_num;
245  } else if (u1) {
246  // Only this is an integer.
247  // NOTE: figure out a way here to use multiply_accumulate(). Most likely we need
248  // a tmp copy, so better profile it first.
249  m_num = m_num*other.m_den + other.m_num;
250  m_den = other.m_den;
251  } else if (u2) {
252  // Only other is an integer.
253  math::multiply_accumulate(m_num,m_den,other.m_num);
254  } else if (m_den == other.m_den) {
255  // Denominators are the same, add numerators and canonicalise.
256  // NOTE: safe if this and other coincide.
257  m_num += other.m_num;
258  canonicalise();
259  } else {
260  // The general case.
261  // NOTE: here dens are different (cannot coincide), and thus
262  // this and other must be separate objects.
263  m_num *= other.m_den;
264  math::multiply_accumulate(m_num,m_den,other.m_num);
265  m_den *= other.m_den;
266  canonicalise();
267  }
268  return *this;
269  }

322  // In-place sub.
323  mp_rational &in_place_sub(const mp_rational &other)
324  {
325  // NOTE: optimisations are possible here if we implement multiply_sub
326  // or do some trickery with in-place negation + multiply_accumulate().
327  // Keep it in mind for future optimisations.
328  const bool u1 = m_den.is_unitary(), u2 = other.m_den.is_unitary();
329  if (u1 && u2) {
330  m_num -= other.m_num;
331  } else if (u1) {
332  m_num = m_num*other.m_den - other.m_num;
333  m_den = other.m_den;
334  } else if (u2) {
335  m_num = m_num - m_den*other.m_num;
336  } else if (m_den == other.m_den) {
337  m_num -= other.m_num;
338  canonicalise();
339  } else {
340  m_num *= other.m_den;
341  // Negate temporarily in order to use multiply_accumulate.
342  // NOTE: candidate for multiply_sub if we ever implement it.
343  m_den.negate();
344  math::multiply_accumulate(m_num,m_den,other.m_num);
