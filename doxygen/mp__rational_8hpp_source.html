152  template <typename Float>
153  void construct_from_interoperable(const Float &x, typename std::enable_if<std::is_floating_point<Float>::value>::type * = nullptr)
154  {
155  if (unlikely(!std::isfinite(x))) {
156  piranha_throw(std::invalid_argument,"cannot construct a rational from a non-finite floating-point number");
157  }
158  // Denominator is always inited as 1.
159  m_den = 1;
160  if (x == Float(0)) {
161  // m_den is 1 already.
162  return;
163  }
164  Float abs_x = std::abs(x);
165  const unsigned radix = static_cast<unsigned>(std::numeric_limits<Float>::radix);
166  int_type i_part;
167  int exp = std::ilogb(abs_x);
168  while (exp >= 0) {
169  i_part += int_type(radix).pow(exp);
170  const Float tmp = std::scalbn(Float(1),exp);
171  if (unlikely(tmp == HUGE_VAL)) {
172  piranha_throw(std::invalid_argument,"output of std::scalbn is HUGE_VAL");
173  }
174  abs_x -= tmp;
175  // Break out if x is an exact integer.
176  if (abs_x == Float(0)) {
177  // m_den is 1 already.
178  m_num = i_part;
179  if (x < Float(0)) {
180  m_num.negate();
181  }
182  return;
183  }
184  exp = std::ilogb(abs_x);
185  if (unlikely(exp == INT_MAX || exp == FP_ILOGBNAN)) {
186  piranha_throw(std::invalid_argument,"error calling std::ilogb");
187  }
188  }
189  piranha_assert(abs_x < Float(1));
190  // Lift up the decimal part into an integer.
191  while (abs_x != Float(0)) {
192  abs_x = std::scalbln(abs_x,1);
193  if (unlikely(abs_x == HUGE_VAL)) {
194  piranha_throw(std::invalid_argument,"output of std::scalbn is HUGE_VAL");
195  }
196  const auto t_abs_x = std::trunc(abs_x);
197  m_den *= radix;
198  m_num *= radix;
199  // NOTE: here t_abs_x is guaranteed to be in
200  // [0,radix - 1], so the cast to unsigned should be ok.
201  // Note that floating-point numbers are guaranteed to be able
202  // to represent exactly at least a [-exp,+exp] exponent range
203  // (see the minimum values for the FLT constants in the C standard).
204  m_num += static_cast<unsigned>(t_abs_x);
205  abs_x -= t_abs_x;
206  }
207  math::multiply_accumulate(m_num,i_part,m_den);
208  canonicalise();
209  if (x < Float(0)) {
210  m_num.negate();
211  }
212  }

236  // In-place add.
237  mp_rational &in_place_add(const mp_rational &other)
238  {
239  // NOTE: all this should never throw because we only operate on mp_integer objects,
240  // no conversions involved, etc.
241  const bool u1 = m_den.is_unitary(), u2 = other.m_den.is_unitary();
242  if (u1 && u2) {
243  // Both are integers, just add without canonicalising. This is safe if
244  // this and other are the same object.
245  m_num += other.m_num;
246  } else if (u1) {
247  // Only this is an integer.
248  // NOTE: figure out a way here to use multiply_accumulate(). Most likely we need
249  // a tmp copy, so better profile it first.
250  m_num = m_num*other.m_den + other.m_num;
251  m_den = other.m_den;
252  } else if (u2) {
253  // Only other is an integer.
254  math::multiply_accumulate(m_num,m_den,other.m_num);
255  } else if (m_den == other.m_den) {
256  // Denominators are the same, add numerators and canonicalise.
257  // NOTE: safe if this and other coincide.
258  m_num += other.m_num;
259  canonicalise();
260  } else {
261  // The general case.
262  // NOTE: here dens are different (cannot coincide), and thus
263  // this and other must be separate objects.
264  m_num *= other.m_den;
265  math::multiply_accumulate(m_num,m_den,other.m_num);
266  m_den *= other.m_den;
267  canonicalise();
268  }
269  return *this;
270  }

323  // In-place sub.
324  mp_rational &in_place_sub(const mp_rational &other)
325  {
326  // NOTE: optimisations are possible here if we implement multiply_sub
327  // or do some trickery with in-place negation + multiply_accumulate().
328  // Keep it in mind for future optimisations.
329  const bool u1 = m_den.is_unitary(), u2 = other.m_den.is_unitary();
330  if (u1 && u2) {
331  m_num -= other.m_num;
332  } else if (u1) {
333  m_num = m_num*other.m_den - other.m_num;
334  m_den = other.m_den;
335  } else if (u2) {
336  m_num = m_num - m_den*other.m_num;
337  } else if (m_den == other.m_den) {
338  m_num -= other.m_num;
339  canonicalise();
340  } else {
