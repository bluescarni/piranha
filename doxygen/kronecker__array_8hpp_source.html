83 template <typename SignedInteger>
84 class kronecker_array
85 {
86  public:
88  typedef SignedInteger int_type;
89  private:
90 #if !defined(PIRANHA_DOXYGEN_INVOKED)
91  static_assert(detail::ka_type_reqs<int_type>::value,"This class can be used only with signed integers.");
92  // This is a 4-tuple of int_type built as follows:
93  // 0. vector of absolute values of the upper/lower limit for each component,
94  // 1. h_min,
95  // 2. h_max,
96  // 3. h_max - h_min.
97  typedef std::tuple<std::vector<int_type>,int_type,int_type,int_type> limit_type;
98  // Vector of limits.
99  typedef std::vector<limit_type> limits_type;
100 #endif
101  public:
103 
107  typedef std::size_t size_type;
108  private:
109  // Static vector of limits built at startup.
110  // NOTE: here we should not have problems when interoperating with libraries that modify the GMP allocation functions,
111  // as we do not store any static piranha::integer: the creation and destruction of integer objects is confined to the determine_limit()
112  // function.
113  static const limits_type m_limits;
114  // Determine limits for m-dimensional vectors.
115  // NOTE: when reasoning about this, keep in mind that this is not a completely generic
116  // codification: min/max vectors are negative/positive and symmetric. This makes it easy
117  // to reason about overflows during (de)codification of vectors, representability of the
118  // quantities involved, etc.
119  static limit_type determine_limit(const size_type &m)
120  {
121  piranha_assert(m >= 1u);
122  std::mt19937 engine(static_cast<unsigned long>(m));
123  std::uniform_int_distribution<int> dist(-5,5);
124  // Perturb integer value: add random quantity and then take next prime.
125  auto perturb = [&engine,&dist] (integer &arg) {
126  arg += (dist(engine) * (arg)) / 100;
127  arg = arg.nextprime();
128  };
129  // Build initial minmax and coding vectors: all elements in the [-1,1] range.
130  std::vector<integer> m_vec, M_vec, c_vec, prev_c_vec, prev_m_vec, prev_M_vec;
131  c_vec.push_back(integer(1));
132  m_vec.push_back(integer(-1));
133  M_vec.push_back(integer(1));
134  for (size_type i = 1u; i < m; ++i) {
135  m_vec.push_back(integer(-1));
136  M_vec.push_back(integer(1));
137  c_vec.push_back(c_vec.back() * integer(3));
138  }
139  // Functor for scalar product of two vectors.
140  auto dot_prod = [](const std::vector<integer> &v1, const std::vector<integer> &v2) -> integer {
141  piranha_assert(v1.size() && v1.size() == v2.size());
142  return std::inner_product(v1.begin(),v1.end(),v2.begin(),integer(0));
143  };
144  while (true) {
145  // Compute the current h_min/max and diff.
146  integer h_min = dot_prod(c_vec,m_vec);
147  integer h_max = dot_prod(c_vec,M_vec);
148  integer diff = h_max - h_min;
149  piranha_assert(diff >= 0);
150  try {
151  // Try to cast everything to hardware integers.
152  (void)static_cast<int_type>(h_min);
153  (void)static_cast<int_type>(h_max);
154  // Here it is +1 because h_max - h_min must be strictly less than the maximum value
155  // of int_type. In the paper, in eq. (7), the Delta_i product appearing in the
156  // decoding of the last component of a vector is equal to (h_max - h_min + 1) so we need
157  // to be able to represent it.
158  (void)static_cast<int_type>(diff + 1);
159  // NOTE: we do not need to cast the individual elements of m/M vecs, as the representability
160  // of h_min/max ensures the representability of m/M as well.
161  } catch (const std::overflow_error &) {
162  std::vector<int_type> tmp;
163  // Check if we are at the first iteration.
164  if (prev_c_vec.size()) {
165  h_min = dot_prod(prev_c_vec,prev_m_vec);
166  h_max = dot_prod(prev_c_vec,prev_M_vec);
167  std::transform(prev_M_vec.begin(),prev_M_vec.end(),std::back_inserter(tmp),[](const integer &n) {
168  return static_cast<int_type>(n);
169  });
170  return std::make_tuple(
171  tmp,
172  static_cast<int_type>(h_min),
173  static_cast<int_type>(h_max),
174  static_cast<int_type>(h_max - h_min)
175  );
176  } else {
177  // Here it means m variables are too many, and we stopped at the first iteration
178  // of the cycle. Return tuple filled with zeroes.
179  return std::make_tuple(tmp,int_type(0),int_type(0),int_type(0));
180  }
181  }
182  // Store old vectors.
183  prev_c_vec = c_vec;
184  prev_m_vec = m_vec;
185  prev_M_vec = M_vec;
186  // Generate new coding vector for next iteration.
187  auto it = c_vec.begin() + 1, prev_it = prev_c_vec.begin();
188  for (; it != c_vec.end(); ++it, ++prev_it) {
189  // Recover original delta.
190  *it /= *prev_it;
191  // Multiply by two and perturb.
192  *it *= 2;
193  perturb(*it);
194  // Multiply by the new accumulated delta product.
195  *it *= *(it - 1);
196  }
197  // Fill in the minmax vectors, apart from the last component.
198  it = c_vec.begin() + 1;
199  piranha_assert(M_vec.size() && M_vec.size() == m_vec.size());
200  for (size_type i = 0u; i < M_vec.size() - 1u; ++i, ++it) {
201  M_vec[i] = ((*it) / *(it - 1) - 1) / 2;
202  m_vec[i] = -M_vec[i];
203  }
204  // We need to generate the last interval, which does not appear in the coding vector.
205  // Take the previous interval and enlarge it so that the corresponding delta is increased by a
206  // perturbed factor of 2.
207  M_vec.back() = (4 * M_vec.back() + 1) / 2;
208  perturb(M_vec.back());
209  m_vec.back() = -M_vec.back();
210  }
211  }
212  static limits_type determine_limits()
213  {
214  limits_type retval;
215  retval.push_back(std::make_tuple(std::vector<int_type>{},int_type(0),int_type(0),int_type(0)));
216  for (size_type i = 1u; ; ++i) {
217  const auto tmp = determine_limit(i);
218  if (std::get<0u>(tmp).empty()) {
219  break;
220  } else {
221  retval.push_back(tmp);
222  }
223  }
224  return retval;
225  }

270  template <typename Vector>
271  static int_type encode(const Vector &v)
272  {
273  const auto size = v.size();
274  // NOTE: here the check is >= because indices in the limits vector correspond to the sizes of the vectors to be encoded.
275  if (unlikely(size >= m_limits.size())) {
276  piranha_throw(std::invalid_argument,"size of vector to be encoded is too large");
277  }
278  if (unlikely(!size)) {
279  return int_type(0);
280  }
281  // Cache quantities.
282  const auto &limit = m_limits[size];
283  const auto &minmax_vec = std::get<0u>(limit);
284  // Check that the vector's components are compatible with the limits.
285  // NOTE: here size is not greater than m_limits.size(), which in turn is compatible with the minmax vectors.
286  for (min_int<decltype(v.size()),decltype(minmax_vec.size())> i = 0u; i < size; ++i) {
287  if (unlikely(boost::numeric_cast<int_type>(v[i]) < -minmax_vec[i] || boost::numeric_cast<int_type>(v[i]) > minmax_vec[i])) {
288  piranha_throw(std::invalid_argument,"a component of the vector to be encoded is out of bounds");
289  }
290  }
291  piranha_assert(minmax_vec[0u] > 0);
292  int_type retval = static_cast<int_type>(boost::numeric_cast<int_type>(v[0u]) + minmax_vec[0u]),
293  cur_c = static_cast<int_type>(2 * minmax_vec[0u] + 1);
294  piranha_assert(retval >= 0);
295  for (decltype(v.size()) i = 1u; i < size; ++i) {
296  retval = static_cast<int_type>(retval + ((boost::numeric_cast<int_type>(v[i]) + minmax_vec[i]) * cur_c));
297  piranha_assert(minmax_vec[i] > 0);
298  cur_c = static_cast<int_type>(cur_c * (2 * minmax_vec[i] + 1));
299  }
300  return static_cast<int_type>(retval + std::get<1u>(limit));
301  }

324  template <typename Vector>
325  static void decode(Vector &retval, const int_type &n)
326  {
327  typedef typename Vector::value_type v_type;
328  const auto m = retval.size();
329  if (unlikely(m >= m_limits.size())) {
330  piranha_throw(std::invalid_argument,"size of vector to be decoded is too large");
331  }
332  if (unlikely(!m)) {
333  if (unlikely(n != 0)) {
334  piranha_throw(std::invalid_argument,"a vector of size 0 must always be encoded as 0");
335  }
336  return;
337  }
338  // Cache values.
339  const auto &limit = m_limits[m];
340  const auto &minmax_vec = std::get<0u>(limit);
341  const auto hmin = std::get<1u>(limit), hmax = std::get<2u>(limit);
342  if (unlikely(n < hmin || n > hmax)) {
343  piranha_throw(std::invalid_argument,"the integer to be decoded is out of bounds");
344  }
345  // NOTE: the static_cast here is useful when working with int_type == char. In that case,
346  // the binary operation on the RHS produces an int (due to integer promotion rules), which gets
347  // assigned back to char causing the compiler to complain about potentially lossy conversion.
348  const int_type code = static_cast<int_type>(n - hmin);
349  piranha_assert(code >= 0);
350  piranha_assert(minmax_vec[0u] > 0);
351  int_type mod_arg = static_cast<int_type>(2 * minmax_vec[0u] + 1);
352  // Do the first value manually.
353  retval[0u] = boost::numeric_cast<v_type>((code % mod_arg) - minmax_vec[0u]);
354  for (min_int<typename Vector::size_type,decltype(minmax_vec.size())> i = 1u; i < m; ++i) {
355  piranha_assert(minmax_vec[i] > 0);
356  retval[i] = boost::numeric_cast<v_type>((code % (mod_arg * (2 * minmax_vec[i] + 1))) / mod_arg - minmax_vec[i]);
357  mod_arg = static_cast<int_type>(mod_arg * (2 * minmax_vec[i] + 1));
358  }
359  }
