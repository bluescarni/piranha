203  template <typename Functor>
204  static void trace_impl(const std::string &str, const Functor &f)
205  {
206  std::lock_guard<std::mutex> lock(m_mutex);
207  bool new_item = false;
208  auto it = m_container.find(str);
209  if (it == m_container.end()) {
210  auto tmp = m_container.insert(std::make_pair(str,boost::any()));
211  piranha_assert(tmp.second);
212  it = tmp.first;
213  new_item = true;
214  }
215  try {
216  f(it->second);
217  } catch (...) {
218  if (new_item) {
219  m_container.erase(it);
220  }
221  throw;
222  }
223  }
